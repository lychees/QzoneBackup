const int N = int(5e4) + 9;
MII last; int P[N]; int f[N];
int n;

int Find(int x){
return P[x] == x ? x : P[x] = Find(P[x]);
}

void Union(int a, int b){
a = Find(a), b = Find(b); assert(a != b);
P[b] = a;
}


int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

while (~scanf("%d", &n)){

CLR(last); REP_1(i, n){
P[i] = i; int ai; RD(ai);
if (CTN(last, ai)) Union(last[ai], last[ai] + 1);
last[ai] = i;

f[i] = INF; for (int jj=1,j=Find(i)-1;sqr(jj)<=f[i];++jj,j=Find(j)-1){
checkMin(f[i], f[j] + sqr(jj));
if (!j) break;
}
}

OT(f[n]);
}
}
