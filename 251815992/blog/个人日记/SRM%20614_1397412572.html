【嵌套 DP】

小环的问题显然就是：
http://acm.hrbust.edu.cn/index.php?m=ProblemSet&a=showProblem&problem_id=1968

。。。原题我们可以方便的解递归式。。。但是这里似乎无论如何都必须保留。。。。
。。。最后的环的出口是否和第一个环的入口是否是同一个颜色这则状态。。。。（好绕口。。）

。。。因此考虑大环间的 DP。。。

f[i][0/1]：表示处理到第 i 个大环，且该环的 out 是否与首环的 in 颜色相同。。。
。。。无法避免要出现下面两个子问题：






我们设：
r1(n) 表示，左右两侧颜色相同。。。中间颜色交错的方案数。。（而这个就是原题。。）
r2(n) 表示，左右两侧颜色不同。。。。



















。。。经过一番努力。。。（用 r2 来表示 r1 似乎更自然些。。。
。。。我只能通过 DP 后找规律。。求一眼看出上面关系的方法！！）


我们得到。。。

Int r2(int n){
if (!n) return 1;
return (pow(m-1, n+1)+((n&1)?-1:+1))/m;
}

Int r1(int n){
if (!n) return 0;
return r2(n)+((n&1)?+1:-1);
}




。。。转移的时候。。讨论 in 和 out 是否一样。。

—— 如果一样。。那么非常简单。。只需要用到 r1 即可。。
—— 如果不一样。。
那么转移分 r1、r2 两种讨论。。

对于。。r1 。。这种转移是不会改变符号的。。
。。对于 r2.。

则有 0 -> 1 （一种。。
1 -> 0。。（m-1 种。
0 -> 0。。（m-2 种。。

设 E[i][j] 表示从到达 i，j 时的期望步数。。。

我们有
E[i][j] =  0                                      | (i, j) = 0
(E[i-1][j] + E[i][j-1]) / 2 + 1 | otherwise
。。（注意 -1 的同时取模。。以下全部都要取模。。不再赘述）。。
。。。首先这个题可以被迭代大发水过。。。

const int N = 109;

DB E[N][N];

class TorusSailing {
public:
double expectedTime(int N, int M, int goalX, int goalY) {

RST(E); DO(6000) REP_2(i, j, N, M) if (i || j){
E[i][j] = (E[(i-1+N)%N][j] + E[i][(j-1+M)%M])/2 + 1;
}

return E[goalX][goalY];
}
};


。。。正解显然要高斯消元。。。把 O(n2) 个变元简化成 O(n) 个。只保留 x 轴 和 y 轴上的两侧。。。。
。。而对于任意的 E[][] 。。。表达成这 O(n) 个 E[][] 的线性组合。。。这点显然可以数学归纳。。。。。

E[i][j] = c0 E[0][0] + c1 E[0][1] + c2 E[0][2] ...
+ cm E[1][0] + c(m+1) E[2][0] + ... + c(m+n) E[n-1][0] + (n+m-1) ...

const int N = 109;

VF c[N][N]; int n, m; VF sum(const VF &a, const VF& b){
VF res = VF(SZ(a), 0);
REP(i, SZ(a)) res[i] = (a[i] + b[i]);
return res;
}

namespace Equ{

VF A[N*2]; void Gauss(){
REP(i, n){
int ii = i; FOR(j, i+1, n) if (fabs(A[j][i]) > fabs(A[ii][i])) ii = j; swap(A[i], A[ii]);

DB t = A[i][i]; A[i][i] = 1;
FOR_1(j, i+1, n) A[i][j] /= t;

REP(j, n) if (i != j){
DB t = A[j][i]; A[j][i] = 0;
FOR_1(k, i+1, n) A[j][k] -= A[i][k]*t;
}
}
}

void init(){

REP_2(i, j, n, m){

c[i][j] = VF(n+m, 0);

if (!i){
c[0][j][j] = j != 0;
}
else if (!j){
c[i][0][m-1+i] = 1;
}
else{
c[i][j] = sum(c[i-1][j], c[i][j-1]);
REP(k, n+m) c[i][j][k] /= 2;
++c[i][j][n+m-1];

}
}

A[0] = VF(n+m, 0), A[0][0] = 1;

FOR(i, 1, m){
A[i]= sum(c[0][i-1], c[n-1][i]);
A[i][n+m-1] += 2, A[i][i] -= 2;
}
FOR(i, 1, n){
A[m-1+i] = sum(c[i-1][0], c[i][m-1]);
A[m-1+i][n+m-1] += 2, A[m-1+i][m-1+i] -= 2;
}

n = n+m-1;
}

DB calc(VF &c){
DB z = 0; REP(i, n) z -= c[i]*A[i][n]; z += c[n];
return z;
}
};
