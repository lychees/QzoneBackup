const int N = int ( 1e5 ) + 9 , M = N * 2 ;


int hd[ N ], suc[ M ], to[ M ], ww[ M / 2 ];

int dep[ N ], par[ N ], n, K;


#define bb to[i]

#define aa to[i^ 1 ]

#define w ww[i/ 2 ]

#define v bb


void dfs( int u = 1 , int p = 0 ){

REP_G (i, u) if ( v != p){

par [ v ] = i;

dep [ v ] = dep [u] + w ;

dfs ( v , u);

}

}

}
