const int N = (1<<16)+9;
int n, _n;
#define n _n
typedef complex<DB> cp; cp P[N];
void FFT(cp P[], const int d = 1){

for(int i=1,j=n/2;i+1<n;++i){
if (i<j) swap(P[i], P[j]); int k = n>>1;
while(j>=k) j-=k, k>>=1; if(j<k) j+=k;
}

for(int lv=1;lv<n;lv<<=1){
cp w0(cos(d*PI/lv),sin(d*PI/lv));
for (int i=0;i<n;i+=lv<<1){
cp w(1, 0); FOR(j, i, i+lv){
cp u = P[j], t = w * P[j+lv];
P[j] = u+t, P[j+lv] = u-t;
w *= w0;
}

}
}
}

int c, cc, nn; DB z;

void dfs_c(int u, int p = -1){
int cz = 0; sz[u] = 1;
REP_G(i, u) if (v != p){
dfs_c(v,u),sz[u] += sz[v];
checkMax(cz,sz[v]);
}
checkMax(cz, nn-sz[u]);
if (cz<=cc)cc=cz,c=u;
}

void dfs0(int u,int p=-1){
sz[u] = 1;dd[u] = d[u];
REP_G(i, u) if (v != p){
d[v]=d[u]+1,dfs0(v,u),sz[u]+=sz[v];

}
}
