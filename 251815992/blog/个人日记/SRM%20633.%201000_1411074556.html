const int PMAX = 46341;
VI P; bitset<PMAX> isC;
#define ii (i*P[j])
void sieve(){

if (!P.empty()) return;

FOR(i, 2, PMAX){
if (!isC[i]) P.PB(i);
for (int j=0;j<SZ(P)&&ii<PMAX;++j){
isC[ii]=1; if (!(i%P[j])) break;
}
}
}
#undef ii

void add_edge(int a, int b){
adj[a].PB(b);
}

void Init(int _n){
n = _n; REP(i, n+n) adj[i].clear();
}

bool Run(){
nn = 0, top = 0; fill(dfn, dfn+n+n, 0); REP(i, n+n) if (!dfn[i]) dfs(i);
REP(i, n) if (bel[i] == bel[i+n]) return 0;
return 1;
}
}

class GCDLCM {

sieve(); int m = type.size();

CLR(PP); REP(i, n) CLR(CC[i]);

PP.insert(*p);
int cc = 1; x /= *p; while (x % *p == 0) ++cc, x/=*p;
CC[i][*p] = cc;
}
}

ECH(p, PP){
