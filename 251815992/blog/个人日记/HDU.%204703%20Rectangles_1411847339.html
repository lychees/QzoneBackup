struct rec{
int l, r;

rec(int l = 0, int r = 0):l(l),r(r){
}

Int operator *(const rec& rhs)const{
return max(min(r, rhs.r) - max(l, rhs.l) + 1, 0);
}
} X[N], Y[N];

Int w[N]; int n;
