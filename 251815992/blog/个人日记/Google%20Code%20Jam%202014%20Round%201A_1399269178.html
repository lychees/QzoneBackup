Problem B. Full Binary Tree

注意 注意 。。。。
Full Binary Tree：正则二叉树。。。
Perfect Binary Tree：满二叉树。。。
http://en.wikipedia.org/wiki/Binary_tree

给定一个无根树，问最少删除多少个结点成为正则二叉树。。。

考虑最多保留多少个结点。。枚举根。。。设 dp[u] 以 u 为根的子树最多可以保留多少个。。那么从子树中最大的两个 dp 值处转移。
。。如果子树小于 2 则。。则需要全部删除。。。。O(n2)。


//}/* .................................................................................................................................. */
const int N = 1009;
VI adj[N]; int n;
#define v (*it)
int dfs(int u, int p = -1){
VI I; ECH(it, adj[u]) if (v != p){
I.PB(dfs(v, u));
}
SRT(I, greater<int>());
return (SZ(I)<=1?0:I[0]+I[1]) + 1;
}

考虑不枚举根。。。 dfs 两次。。。第二次的时候从父亲结点 O(1)
local update 出以任意结点为根的答案。。。 。那么显然需要记录 3 个 instead of 、2 个。。


//}/* .................................................................................................................................. */

const int N = 1009;

VI adj[N]; VII I[N]; int n;

int f(VII& I){
SRT(I, greater<PII>()); if (SZ(I) > 3) I.resize(3);
return (SZ(I)<=1?0:I[0].fi+I[1].fi) + 1;
}

#define v (*it)
int dfs1(int u = 0, int p = -1){
I[u].clear(); ECH(it, adj[u]) if (v != p){
I[u].PB(MP(dfs1(v, u), v));
}
return f(I[u]);
}

int dfs2(int u = 0, int p = -1){
int z = f(I[u]); ECH(it, adj[u]) if (v != p){
VII II = I[u]; REP(i, SZ(II)) if (II[i].se == v) ERS(II, i--);
I[v].PB(MP(f(II), u)), checkMax(z, dfs2(v, u));
}
return z;
}

#undef v
