ll merge_count(vector<int> &a) {
int n = a.size();
if (n <= 1) return 0;
vector<int> b(a.begin(), a.begin() + n / 2);
vector<int> c(a.begin() + n / 2, a.end());
ll res = 0;
res += merge_count(b);
res += merge_count(c);
for (int ai = 0, bi = 0, ci = 0; ai < n; ++ai) {
if (ci == (int)c.size() || (bi < (int)b.size() && b[bi] <= c[ci])) {
a[ai] = b[bi++];
} else {
a[ai] = c[ci++];
res += n / 2 - bi;
}
}
return res;
}
