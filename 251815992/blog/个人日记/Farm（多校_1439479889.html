REP(i, n+1) RD(buy[i], sell[i]);

LL z = 0; int cc = 0;

cz = 0, op = 0; Q[0] = MP(buy[0], c);

REP_1(i, n){

int t = a[i]; while (t){
int d = min(t, Q[cz].se);  Q[cz].se -= d; t -= d;
z += (LL)Q[cz].fi * d; if (!Q[cz].se) ++cz;
}

int cc = 0; while (cz <= op && Q[op].fi >= buy[i]) cc += Q[op--].se;
Q[++op] = MP(buy[i], cc + a[i]);

LL s = 0; while (cz < op && Q[cz].fi < sell[i]){
z += (LL)Q[cz].fi * Q[cz].se;
s += Q[cz++].se;
}
if (s){
--cz;
Q[cz].fi = sell[i]; Q[cz].se = s;
z -= (LL)sell[i] * s;
}
}
OT(z);
}

const int N = 309;

int W[N][N], lx[N], ly[N], p[N], slack[N];
bool vx[N], vy[N];
int n, m, total;

struct rec{
int tp, lv, atk;
void in(){
RD(tp, lv, atk);
total += atk;
}
} V[N];
VI X, Y;

void add(int a, int b, int atk){
int w = atk - V[a].atk - V[b].atk; if (w > 0){
if (V[a].tp) swap(a, b);
checkMax(W[a][b], w);
}
}

void init(){
total = 0; X.clear(); Y.clear(); RD(n, m); RST(W);
REP_1(i, n) V[i].in();

REP(i, m){
int lv, atk, r, a, b; RD(lv, atk, r); if (r == 2){
RD(a, b); add(a, b, atk);
}
else if (r == 1){
RD(a); REP_1(b, n) if (V[b].tp != V[a].tp && V[a].lv + V[b].lv == lv){
add(a, b, atk);
}
} else{
REP_1(a, n) REP_1(b, n) if (V[b].tp != V[a].tp && V[a].lv + V[b].lv == lv){
add(a, b, atk);
}
}
}
}

#define w(x, y) (lx[x] + ly[y] - W[x][y])

bool dfs(int x){
vx[x] = true; REP_1(y, n) if (!vy[y]){
if (!w(x, y)){
vy[y] = true; if(p[y]==-1||dfs(p[y])){
p[y] = x;
return true;
}
}
else {
checkMin(slack[y], w(x, y));
}
}
return false;
}

int KM(){
FLC(p, -1); RST(lx, ly);

REP_2_1(i, j, n, n) checkMax(lx[i], W[i][j]);

REP_1(x, n){
FLC(slack, 0x7f); while (true){
RST(vx, vy); if (dfs(x)) break;

int delta = INF;
REP_1(i, n) if (!vy[i]) checkMin(delta, slack[i]);

REP_1(i, n){
if (vx[i]) lx[i] -= delta;
if (vy[i]) ly[i] += delta; else slack[i] -= delta;
}
}
}
int z = 0;  REP_1(y, n) z += W[p[y]][y];
return z;
}


int main(){
#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
#endif
Rush{
init();
printf("%d\n", total + KM());
}
}
