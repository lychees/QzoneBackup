int det(PII p1,PII p2) {return p1.fi*p2.se-p2.fi*p1.se;}
int det(PII p0,PII p1,PII p2){return det(MP(p1.fi-p0.fi,p1.se-p0.se),MP(p2.fi-p0.fi,p2.se-p0.se));}

const int N = 71;

PII ans; int A[N][N], B[N][N];
int W[N][N], lx[N], ly[N], p[N], delta;
bool vx[N], vy[N]; int n;

#define w(x, y) (lx[x] + ly[y] - W[x][y])

bool dfs(int x){
vx[x] = true; REP(y, n) if (!vy[y]){
if (!w(x, y)){
vy[y] = true; if(p[y]==-1||dfs(p[y])){
p[y] = x;
return true;
}
}
else {
checkMin(delta, w(x, y));
}
}
return false;
}

void KM(PII &z){

REP_2(i, j, n, n) W[i][j] = -W[i][j];

FLC(p, -1); RST(lx, ly); REP_2(i, j, n, n) checkMax(lx[i], W[i][j]);

REP(x, n){
while (1){
RST(vx, vy); delta = INF; if (dfs(x)) break;

REP(i, n){
if (vx[i]) lx[i] -= delta;
if (vy[i]) ly[i] += delta;
}
}
}

REP(i, n) z.fi += A[p[i]][i], z.se += B[p[i]][i]; //cout << z.fi << " " << z.se << endl;
if (z.fi * z.se < ans.fi * ans.se) ans = z;
}

void Init(){
RD(n); REP_2(i, j, n, n) RD(A[i][j]); REP_2(i, j, n, n) RD(B[i][j]);
}

void gao(PII&l, PII&r){
PII p(r.fi-l.fi,r.se-l.se); REP_2(i, j, n, n) W[i][j] = det(p, MP(A[i][j], B[i][j]));
PII m; KM(m); if (!det(l,m,r)) return; gao(l, m), gao(m, r);
}

void XKM(){
ans = MP(INF, 1); PII l, r;
REP_2(i, j, n, n) W[i][j] = A[i][j]; KM(l);
REP_2(i, j, n, n) W[i][j] = B[i][j]; KM(r);
if (l.fi!=r.fi||l.se!=r.se) gao(l,r);
printf("%d\n", ans.fi * ans.se);
}
