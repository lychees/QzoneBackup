这个题十分有趣。。。10^6 的规模。。。显然直接主席树是不行的。。
。。需要深入分析。。


算法一：

考虑一条链。。。假设询问是 a, b。。（a 在左边。。）
。。。显然我们只要预处理出。。每个结点向左和向右没有出现的最小值 l[] 和 r[]
。。。那么答案就是 min(l[a], r[b])。。。

。。。我们以 1 为根。。。依次找到包含 2,3,4 ... 的生成子图。。。显然在某一刻比如包含 x 的时候。。。该子图不再是一条链。。
。。。那么所有包含了 [1..x) 的路径。。都不可能包含 x。。。

因此问题就转化成了链上的问题。。。如果包含了这条链 那么答案是 x。。。否则在这条链上询问。。


算法二：

设 no[u] 表示，u 所在分支，不在 0->u 路径上的最小值。
（。。分支表示在 1 结点的哪个孩子中。。。）

。。。那么对于询问 a, b。。答案等于
。。。min(no[a], no[b],  非 a、b 所在分支的最小值)。。。

。。。我们用 u 所在分支的最小值代表整个分支，记做 bel[u] 。。。
。。。最后一项。。我们只要求出根结点向下 mn[1][3] 最小、次小、次次小。。。
。。。和 bel[a]，bel[b] 比较就可以了。。。


no[v] 的值由两部分组成。。
。。v 所在子树。。
。。v 之上。。

。。先 DP 出后者。再用 mn[] 把前者补上。。

const int N = int(1e6) + 9;

int n, q, hd[N], to[N << 1], suc[N << 1];
int Q[N], fa[N], depth[N], bel[N], ch[N][4], path[N];

#define aa to[i^1]
#define bb to[i]
#define v bb

int query(int a, int b){
if (a > b) swap(a, b); if (a && bel[a] == bel[b]) return 0;
int i = 0; while (ch[0][i] == bel[a] || ch[0][i] == bel[b]) i ++;
return min(a ? path[a] : INF, path[b], ch[0][i]);
}

int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

while (~scanf("%d%d", &n, &q)){

fill(hd, hd+n, 0); FOR(i, 2, n<<1){
RD(aa, bb), --aa, --bb;
suc[i] = hd[aa], hd[aa] = i++;
suc[i] = hd[aa], hd[aa] = i;
}

int cz = 0, op = 0; Q[op++] = 0; fa[0] = -1; while (cz < op){
int u = Q[cz++]; REP_G(i, u) if (v != fa[u]){
fa[v] = u, depth[v] = depth[u] + 1;
Q[op++] = v;
}
}

FLC(ch, 0x3f); DWN(i, n, 0){
int u = Q[i]; bel[u] = min(u, ch[u][0]);
int p = fa[u]; if (u){
ch[p][3] = bel[u];
sort(ch[p], ch[p] + 4);
}
}

op = 0; REP_G(i, 0){
path[v] = INT_MAX;
Q[op ++] = v;
}

path[0] = INT_MAX;

cz = 0; while (cz < op){
int u = Q[cz++]; REP_G(i, u) if (v != fa[u]){
path[v] = min(path[u], ch[u][bel[v]==ch[u][0]]);
bel[v] = bel[u];
Q[op++] = v;
}
checkMin(path[u], ch[u][0]);
}

la = 0; DO(q){
int a, b; RD(a, b); a ^= la, b ^= la; --a, --b;
la = query(a, b) + 1;
OT(la);
}
}
}
