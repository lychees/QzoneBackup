int find(int x) {
if (fa[x] == x)
return x;
return fa[x] = find(fa[x]);
}

bool unite(int a, int b) {
a = find(a), b = find(b);
if (a == b)
return false;
fa[a] = b;
size[b] += size[a];
return true;
}
