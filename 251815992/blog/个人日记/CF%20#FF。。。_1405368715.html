const int N = int(3e5)+9;

Int fib[N], sib[N];

Int ss[4*N]; pair<Int,Int> bj[4*N];
int n, a, b, c;

#define lx (x<<1)
#define rx (lx|1)
#define ml (l+r>>1)
#define mr (ml+1)
#define lc lx,l,ml
#define rc rx,mr,r

void upd(int x){
ss[x] = ss[lx] + ss[rx];
}

void inc(int x, int l1, Int b0, Int b1){
ss[x] += b0*(sib[l1-1]+1) +b1*sib[l1];
bj[x].fi += b0, bj[x].se += b1;
}

void rls(int x, int l, int r){
if (bj[x].fi || bj[x].se){
inc(lx, ml-l, bj[x].fi, bj[x].se); bj[x] = MP(bj[x].fi*fib[ml-l]+bj[x].se*fib[mr-l], bj[x].fi*fib[mr-l]+bj[x].se*fib[mr-l+1]);
inc(rx, r-mr, bj[x].fi, bj[x].se); bj[x] = MP(0, 0);
}
}

void Modify(int x = 1, int l = 1, int r = n){
if (b < l || r < a) return;
if (a <= l && r <= b){
inc(x, r-l, fib[c], fib[c+1]);
c += r-l+1;
}
else{
rls(x, l, r);
Modify(lc), Modify(rc);
upd(x);
}
}

Int Query(int x = 1, int l = 1, int r = n){
if (b < l || r < a) return 0;
if (a <= l && r <= b) return ss[x];
rls(x,l,r); return Query(lc) + Query(rc);
}

void Build(int x = 1, int l = 1, int r = n){
if (l == r){
ss[x] = RD();
}
else{
Build(lc), Build(rc);
upd(x);
}
}

int cmd; DO(m){
if (RD(cmd, a, b) == 1){
c = 1, Modify();
}
else{
OT(Query());
}
}
