#include <iostream>
#include <cmath>

using namespace std;
const int N = 250;
const double eps = 0.8;

int h,w;
double matrix[N][N];

bool isZero(double x){
return fabsl(x) < eps;
}

int getRank(){
int now_row = 0;
for(int i = 0; i < h; i++){
int flag = -1;
for(int j = now_row; j < w; j++){
if(!isZero(matrix[j][i])){
for(int k = 0; k < h; k++) {
swap(matrix[j][k], matrix[now_row][k]);
}
flag = 0;
break;
}
}
if(!flag){
for(int j = now_row + 1; j < w; j++){
double tp= matrix[j][i] / matrix[now_row][i];
for(int k = i; k < h; k++){
matrix[j][k] -= matrix[now_row][k] * tp;
}
}
now_row++;
}
}
int rank = 0;
for(int i = 0; i < w; i++){
for(int j = 0; j < h; j++){
if(!isZero(matrix[i][j])){
rank++;
break;
}
}
}
return rank;
}

int main(){
cin >> h >> w;
for (int i = 0; i < h; i++){
for (int j = 0; j < w; j++){
cin >> matrix[i][j];
}
}
cout << getRank() << endl;
return 0;
}
