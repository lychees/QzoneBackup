const int N = int(1e5)+9, M = 2*N;

struct node{

static node* NIL; node *c[2], *p;
int w1, w2, w3, w4[2], d0; bool r0;

#define NIL node::NIL
#define l c[0]
#define r c[1]
#define lx x->l
#define rx x->r
#define px x->p
#define ly y->l
#define ry y->r
#define py y->p

void reset(){
l = r = p = NIL;
w4[0] = w4[1] = w2 = d0 = r0 = 0;
w3 = INF;
}

inline node(){
reset();
}


inline void upd(){
if (this == NIL) return;
w2 = max(l->w2, w1, r->w2), w3 = min(l->w3, w1, r->w3);
w4[0] = max(l->w4[0], r->w4[0], max(l->w2, w1) - min(w1, r->w3));
w4[1] = max(l->w4[1], r->w4[1], max(r->w2, w1) - min(w1, l->w3));
}

inline void rev(){
r0 ^= 1, swap(l, r);
swap(w4[0], w4[1]);
}

inline void inc(int d){
if (this == NIL) return;
w1 += d; w2 += d, w3 += d;
d0 += d;
}


inline void rls(){
if (this == NIL) return;
if (r0){
l->rev(), r->rev();
r0 = 0;
}
if (d0){
l->inc(d0), r->inc(d0);
d0 = 0;
}
}

// 旋转

inline int sgn(){return p->l==this?0:p->r==this?1:-1;}
inline void setc(int d,node*x){c[d]=x,px=this;}

inline void rot(int d){
node *y = p, *z = py; if (~y->sgn()) z->setc(y->sgn(), this); else p = z;
y->setc(!d, c[d]), setc(d, y), y->upd();
}

inline void rot(){rot(!sgn());}
inline void zag(){rot(0);}
inline void zig(){rot(1);}

inline node* splay(){
fix(); while (~sgn()) rot(); upd();
return this;

inline node* acs(){
node *x = this, *y = NIL; do{
x->splay();
rx = y, x->upd();
y = x, x = px;
} while (x != NIL);
return splay();
}
