树形 DP。。

搞两次。。设
g[] 为考虑子树时，被染色的概率。
f[] 为再考虑父亲时，被染色的概率。


定义两个算符。。

子树合并 ：
DB mm(DB x, DB y){
return x + y - x*y;
}

子树分拆：（逆运算）
DB dd(DB x, DB y){
if (!sgn(1-y)) return 1;
return (x-y) / (1-y);
}

g[] 转移显然。。
f[] 。。

f[1] = g[1]。。
对于其他点。。考虑从父亲那条边染色过来。。 我们有 f[u] 。。
但是需要容斥掉从 v 传递回去的情况。。
也就是用到前面的子树分拆。。


//}/* .................................................................................................................................. */

const int N = int(5e5) + 9, M = 2*N;

int hd[N], suc[M], to[M]; DB _pp[N];
DB g[N], f[N], res; int n, m;

#define aa to[i^1]
#define bb to[i]
#define v bb
#define pp _pp[i/2]

DB mm(DB x, DB y){
return x + y - x*y;
}

DB dd(DB x, DB y){
if (!sgn(1-y)) return 1;
return (x-y) / (1-y);
}

void dfs_g(int u = 1, int p = 0){
REP_G(i, u) if (v != p){
dfs_g(v, u);
g[u] = mm(g[u], g[v]*pp);
}
}

void dfs_f(int u = 1, int p = 0){
res += f[u]; REP_G(i, u) if (v != p){
f[v] = mm(g[v], dd(f[u], g[v]*pp)*pp);
dfs_f(v, u);
}
}

int main(){

FOR_C(i, 2, RD(n)<<1){
RD(aa, bb); pp = (DB) RD() / 100;
suc[i] = hd[aa], hd[aa] = i, ++i;
suc[i] = hd[aa], hd[aa] = i;
}

REP_1(i, n) g[i] = (DB) RD() / 100; dfs_g(), f[1] = g[1], dfs_f();
OT(res);
}
