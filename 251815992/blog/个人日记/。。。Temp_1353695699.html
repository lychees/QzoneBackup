const int N = 100009;
int P[N], R[N], nn;
int n, m, p;

void Make(int x){
R[x] = 0, P[x] = x;
}

void Init(){
REP(i, n) Make(i);
}

int Find(int x){
return P[x] = P[x] == x ? x : Find(P[x]);
}

void Union(int x, int y){
x = Find(x), y = Find(y);
if (x == y) return;
if (R[x] < R[y]) P[x] = y;
else {
if (R[x] == R[y]) ++R[x];
P[y] = x;
}
--nn;
}

int main(){

#ifndef ONLINE_JUDGE
//freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

while (scanf("%d%d%d", &n, &m, &p) != EOF){
Init(); DO(m) Union(RD(), RD());
DO(p) puts(Find(RD()) == Find(RD()) ? "Yes" : "No");
}
}
