这个题的推广。。。

可以如法炮制得到一个 O(n4) DP。。。。

const int N = 209;

DB dp[2][N][N][2*N], pl, pr, ps;
int n;

DB f(int n){

int p = 0, q = 1; RST(dp[p]); dp[p][0][0][0+N] = 1;

REP(i, n){
swap(p, q); RST(dp[p]);
#define v dp[p]
#define u dp[q][l][r][j+N]
#define ll (-j==l?l+1:l)
#define rr (j==r?r+1:r)

FOR_1(l, 0, i) FOR_1(r, 0, i) FOR_1(j, -l, r) if (u){
//v[j][k+N] += u*ps;
v[ll][r][j-1+N] += u*pl;
v[l][rr][j+1+N] += u*pr;
}
}

swap(p, q);
DB res = 0; FOR_1(l, 0, n) FOR_1(r, 0, n) FOR_1(j, -l, r) res += u*(l+r+1);
return res;
}


。。。但是这样 N = 509 数组都开不下。。。。。。
我们注意到很多状态本质是一样的。。可以合并。。。
只需要记录当前向左右伸展的总长度，以及距离左端点的位置就好了。。。




//}/* .................................................................................................................................. */

const int N = 509;

DB dp[2][2*N][N], pl, pr, ps;
int n;

DB f(int n){

int p = 0, q = 1; RST(dp[p]); dp[p][1][0] = 1; REP_1(i, n){

swap(p, q); RST(dp[p]);

#define v dp[p]
#define u dp[q][l][j]

REP_1(l, i) REP(j, l) if (u){
int ll, jj;
ll = l, jj = j; if (!jj) ++ll; else --jj; v[ll][jj] += u*pl;
ll = l, jj = j+1; if (jj==l) ++ll; v[ll][jj] += u*pr;
}
}

swap(p, q);
DB res = 0; REP_1(l, n+1) REP(j, l){
res += u*l;
}
return res;
}

class RedPaint {
public:
double expectedCells(int N) {
pl = 0.5, pr = 0.5;
return f(N);
}
};

发现有一种非常厉害的搞发。。。
dp[i][j]： 表示走了 i 步。。当前 rightmost 位置在 j。。。




const int N = 509;

DB dp[2][N], pl, pr, ps;
int n;

DB f(int n){
int p = 0, q = 1; RST(dp[p]); dp[p][0] = 1; REP_1(i, n){
swap(p, q); RST(dp[p]);
FOR_1(j, 0, i) dp[p][j] = (dp[q][max(0, j-1)] + dp[q][j+1])/2;
}
DB z = 0; REP_1(i, n) z += dp[p][i]*i;
return 2*z+1;
}

class RedPaint {
public:
double expectedCells(int N) {
pl = 0.5, pr = 0.5;
return f(N);
}
}
