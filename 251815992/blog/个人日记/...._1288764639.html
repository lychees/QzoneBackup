int gcd( int a, int b){

if (a < b) swap (a, b);

int t = a - b;


while (t!= 0 ) {

a -= b;

if (a < b) swap (a, b);

t = a - b;

}

return a;

}


int lcm( int a, int b){

return a * b / gcd (a, b);

}
