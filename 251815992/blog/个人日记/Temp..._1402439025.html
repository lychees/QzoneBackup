首先做一下 HDU 4344。。。网上搜了一圈回去重看了一下算法导论。。
发现我之前 Miller_Rabin 测试写的是简化版。。。


完整版需要加一个步骤。。就是试图寻找 1 的非平凡平方根。。

inline bool Miller_Rabin(int p,LL n){
int t=0; LL u=n-1; while(!(u&1)) ++t,u>>=1;
LL x=poww(p,u,n); if(x==1) return 1; DO(t){
if(x==n-1) return 1;
x=pdtt(x,x,n);
}
return 0;
}

这里  u = 2^t x 分解成。。t 可以看成其二进制分解后末尾 0 的个数。。那么这体现在快速幂中就是初始时的 t 个连乘。。
。。我们把这个步骤分离出来。。

最后做 t 个判断。。
if(x==n-1) return 1;

CLRS 上这里写了三个判断。。但是实际上用这一个就全部包含了。

inline bool isPrime(LL n){
static const int P[]={2,3,5,7,11,13,17,19,23}, Pn = 9;
if(find(P,P+Pn,n)!=P+Pn) return 1; if(n==1||!(n&1)) return 0;
REP(i, Pn) if(!Miller_Rabin(P[i],n)) return 0;
return 1;
}

然后根据 GYZ 的说法取前 9 个素数可以保证在 10^18 不出错。。。

void rho(LL n){

if(isPrime(n)){
fac[n]++;
return;
}

int c=1; while(1){
LL x1=1,x2=1; int i=1,k=2; while(1){
x1=pdtt(x1,x1,n)+c;
LL d=gcd(abs(x1-x2),n);
if(d!=1&&d!=n){
rho(d),rho(n/d);
return;
}
if(x1==x2) break;
if(++i==k) k<<=1,x2=x1;
}
++c;
}
}

这里有一个非常高端的 a*b%c 代码。。。用 long double 来替代 128 位乘。。我觉得非常有效。

inline LL pdtt(LL x,LL y,LL z){
return (x*y-(LL)(((long double)x*y+0.5)/z)*z+z)%z;
}
