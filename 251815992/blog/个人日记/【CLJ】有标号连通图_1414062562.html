const int N = 16;

Int C[N][N];
Int A[1<<15][N]; // For all cc <= j

int n; bool G[N][N]; int Edge[1<<15];
Int res[N];

int count_edge(int a, int s){
int z = 0; REP(i, n) if (_1(s, i)) z += G[i][a];
return z;
}


int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

REP(i, N){
C[i][0] = 1;
REP_1(j, i) C[i][j] = C[i-1][j-1] + C[i-1][j];
}

RD(n); Rush{
int a, b; RD(a, b); --a, --b;
G[a][b] = G[b][a] = 1;
}

FOR(s, 1, _1(n)){
int i = count_bits(s); REP_1(j, i-1){
A[s][j] = 0; int _s = s^low_bit(s);
REP_SS(ss, _s) if (i-count_bits(ss) <= j){
int s_ = s^ss|low_bit(s);
A[s][j] += (A[s_][count_bits(s_)]-A[s_][count_bits(s_)-1]) * A[ss][min(count_bits(ss), j)];
}
if (i == n) res[j] += A[s][j];
}
Edge[s] = Edge[s^low_bit(s)] + count_edge(low_idx(s), s^low_bit(s));
A[s][i] = pow(2, Edge[s]);
if (i == n) res[i] += A[s][i];
}

DWN_1(i, n, 2){
res[i] -= res[i-1];
