/* -- Start from here -- */

const int N = 2509;

DB f[N][N]; // 还剩 i 张牌。。当前知道 j 张。

#define u f[i][j]
#define q (2*i-j)
#define p1 ((DB)j/q)  // 翻到一个已知的
#define p0 (1-p1) // 翻到一个未知的
#define p02 ((q-1)?p0/(q-1):0) //翻到一个未知的后、恰好翻到和它一样的
#define p01 (p02*j)  // 翻到一个未知的后、紧接着翻到一个已知的
#define p00 (p0-p01-p02)  // 翻到一个未知的后、紧接着又翻到一个未知的

class PerfectMemory {
public:
double getExpectation(int n, int m) {

n = n*m/2; RST(f); REP_1(i, n) DWN_1(j, i, 0){
u = 1; if (j) u += p1*f[i-1][j-1];
u += p02*f[i-1][j]+p01*(f[i-1][j]+1)+p00*f[i][j+2];
}

return f[n][0];
}
};
