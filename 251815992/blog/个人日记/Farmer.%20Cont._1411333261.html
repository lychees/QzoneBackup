。。。枚举每一行。。。统计所有以这行为底的矩形。。。
。。方法是左右两个栈。。求出以每个位置为中心的极大子矩形。。。然后去掉重复的部分。

至于怎么去重。。


比如一个 “凸” 的形状。。。。

。。1 1 3 3 1 1。。。

中间 1*2 的部分。会被两个极大子矩形同时计数到。。。。
。。。于是我们在扫描中间那个极大子矩形的时候。。就要把这个部分去掉。。（有点类似后缀数字里的 height[]。。。）

set<pair<PII, int> > H; REP_1(i, m) if(h[i]){
int ww = r[i] - l[i], hl = max(h[r[i]], h[l[i]-1]), hr = h[i];
if (CTN(H, MP(MP(l[i],r[i]),h[i]))) continue; H.insert(MP(MP(l[i],r[i]),h[i]));
--s[hl][ww],++s[hr][ww];
}

这里 l[i], r[i] 是极大子矩形向两侧最远延伸出的距离。
。。 h[i] 是高度。。。hl 就是会多出来的部分。。

const int N = int(1.1e3) + 9;

int s[N][N], f[N][N]; char Map[N][N];
int h[N], l[N], r[N];
int n, m;

int main(){

#ifndef ONLINE_JUDGE
//freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

RD(n, m); REP_1(i, n) RS(Map[i]+1);


h[0] = 0, h[m+1] = 0; REP_1(i, n){
REP_1(j, m) h[j] = f[i][j] = (Map[i][j] == '1' ? f[i-1][j] + 1 : 0);

stack<int> sta;

#define s sta
s.push(0); REP_1(i, m){
if (h[i]){
while (h[i] <= h[s.top()]) s.pop();
l[i] = s.top()+1;
}
else{
CLR(s);
}
s.push(i);
}

CLR(s); s.push(m+1); DWN_1(i, m, 1){
if (h[i]){
while (h[i] <= h[s.top()]) s.pop();
r[i] = s.top();
}
else{
CLR(s);
}
s.push(i);
}
#undef s

set<pair<PII, int> > H; REP_1(i, m) if(h[i]){
int ww = r[i] - l[i], hl = max(h[r[i]], h[l[i]-1]), hr = h[i];
if (CTN(H, MP(MP(l[i],r[i]),h[i]))) continue; H.insert(MP(MP(l[i],r[i]),h[i]));
--s[hl][ww],++s[hr][ww];
}
}

DWN_1(i, n, 1) DWN_1(j, m, 1) s[i][j-1] += s[i][j];
DWN_1(i, n, 1) DWN_1(j, m, 1) s[i][j-1] += s[i][j];
DWN_1(i, n, 1) DWN_1(j, m, 1) s[i-1][j] += s[i][j];

REP_1(i, n){
REP_1(j, m) printf("%d ", s[i][j]);
puts("");
}
}
