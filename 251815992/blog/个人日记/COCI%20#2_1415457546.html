const int N = int(700) + 9;
const DB safe = -1, bad = -2;

int h[N][N], v[N][N];
int n, res, cpt;
set<DB> vis[N][N]; int vis2[N][N], Time;

bool inGrid(int x, int y){
return x >= 0 && y >= 0 && x < n && y < n;
}

DB f(int x1, int y1, int x2, int y2){
if (h[x1][y1] == h[x2][y2]){
if (v[x1][y1] == v[x1][y2]) return safe;
return 0;
}
else{
if (v[x2][y2] == v[x1][y1]) return bad;
DB t = (DB)(h[x1][y1] - h[x2][y2]) / (DB)(v[x2][y2] - v[x1][y1]);
if (sgn(t) == -1) return bad;
return t;
}
}

DB comb(DB a, DB b){
//cout << a << " " << b << ": ";
if (a == bad || b == bad) return bad;
if (a == safe) return b;
if (b == safe) return a;
return sgn(a, b) ? bad : a;
}

int dfs(int x, int y, DB t){
//if (t != safe)
vis[x][y].insert(t);
//cout << x << " " << y << " " << t << endl;
map<DB, int> dp;

vis2[x][y] = Time;
REP(d, 4){
int xx = x + dx[d], yy = y + dy[d];
if (!inGrid(xx, yy) || vis2[xx][yy] == Time) continue;
DB tt = comb(t, f(x,y,xx,yy));
if (tt == bad) continue;
if (!CTN(vis[xx][yy], tt)){
dp[tt] += dfs(xx, yy, tt);
}
}
vis2[x][y] = 0;
int z = 0;ECH(it, dp) checkMax(z, it->se);
return z + 1;
}
