。。。我们可以从小的数开始从两边开始填。。
。。不考虑限制的话是 3^(n-1) 。。（都填左边，都填右边，两边各填一个。。。

考虑限制的话需要区间 dp。。在 dp 到那个位置的时候。。判断一下先后关系即可。。。

const int N = 109;

struct rec{
int p; string s;
void in(){
cin >> s; RD(p);
}
bool eq(int l, int r, int b){
return p == b;
}
bool le(int l, int r, int b){
return l <= p && p <= r && p != b;
}
bool ge(int l, int r, int b){
return p < l || r < p;
}

bool ok(int l, int r, int b){
if (s == "=") return eq(l, r, b);
if (s == "<") return le(l, r, b);
if (s == ">") return ge(l, r, b);
if (s == "<=") return le(l, r, b) || eq(l, r, b);
if (s == ">=") return ge(l, r, b) || eq(l, r, b);
return true;
}
}; vector<rec> R[N];

LL F[N][N];
int n;

bool ck(int l, int r, int a, int b){ //
ECH(it, R[a]) if (!it->ok(l, r, b)) return false;
ECH(it, R[b]) if (!it->ok(l, r, a)) return false;
return true;
}

LL f(int l, int r){
LL &z = F[l][r];
if (z == -1){
if (l+1 == r){
z = ck(l,r,l,r);
}
else{
z = 0;
if (ck(l, r, l, l+1)) z += f(l+2, r);
if (ck(l, r, r, r-1)) z += f(l, r-2);
if (ck(l, r, l, r)) z += f(l+1, r-1);
}
}
return z;
}

int main() {

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

int m; RD(n, m); DO(m){
int x; rec r; RD(x); r.in();
if (x == r.p){
if (r.s.find('=') == string::npos){
puts("0");
return 0;
}
}
else{
R[x].PB(r);
}
}

FLC(F, -1);
cout << f(1, 2*n) << endl;
}
