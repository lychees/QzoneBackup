。。超过这个范围的话。。。就会沿着 fail 树的 par 指针跑到上一个结点中去了。。（逆序的后缀树。）

——————

。。现在我们以 fail 树的视角来考察这些结点。。

每个结点表示了某个子串，以及及该子串的最长的一部分后缀所组成的集合。。。（准确的说是这些子串的逆构成了一个后缀树。）

。其中最长的那个子串的长度是 len[u]，最短的子串长度是 len[p]+1。。
。所包含的子串数目。。是 len[u] - len[p]。。。（对应后缀树的边长）
。我们称一个状态中最长的子串为这个状态的特征子串。。。

——————
。。。其中所有叶子结点表示的特征子串恰好在这个字符串中仅出现一次。。（。。。满足这个要求的肯定是一些前缀。。。）
（参见 识别子串 问题。。。）

。。。而一个结点到根的路径。。恰好构成该结点所包含的特征子串的所有后缀。。。



1. 对于原问题。。。我们主要任务就是。。
。。。。 得到每个结点（所表示的子串）。。可以匹配哪些字符串。
当然由于原字符串数目过多。。要对每个结点都得到具体匹配哪些字符串 并不现实 。


2. tire 也是具有一定定位能力的。。。它仅能得知每个结点是否是某个字符串的一个前缀。。。。。
。。。而根据之前的讨论。。我们找到这些前缀在 SAM 中的对应结点。。显然所有这些结点到根的路径的并所包含的结点。。
。。。恰好包含了关联于这个字符串的所有子串。。在 SAM 中的状态。。。

3. 我们可以使用 tarjan lca 来求树上链的并。。。这样我们就可以统计出。。 每个结点。。。匹配了多少字符串。。
。。如果这个值 >= K。。。那么我们就对这个结点打上待统计的标记。。 cnt[u] = len[u] - len[p]。。


4. 回到第二步。。我们知道每个结点是否是某个字符串的前缀。。而这个结点到根的路径上所有的结点。。构成了这个前缀的所有后缀。
。。。显然这些后缀也应该属于这个字符串。。。于是我们可以用 dfs() 。得到由根到这个结点的路径上所有经过结点的 cnt 和。。。。
。。。。把所有前缀都访问到就可以统计所有子串的贡献。。。。总得复杂度是线性的。。。。。。

const int N = 200009, LN = 24, Z = 26;

char buf[N/2]; LL ans[N/2];
int TtoM[N/2], MtoT[N], nn, K;

namespace Trie{
int trans[N/2][Z]; VI b[N/2];
int tot;

#define v trans[u][c]

int new_node(){
return tot++;
}

void Insert(int id){
RS(buf); int u = 0; REP_S(cur, buf){
int c = *cur - 'a';
if (!v) v = new_node();
b[u = v].PB(id);
}
}

void Init(){
tot = 0, new_node();
REP(i, nn) Insert(i);
}
}

namespace SAM{

int trans[N][Z], par[N], len[N], cnt[N], tot;
VI adj[N]; int P[N], R[N], A[N], L[N/2];

#define p par[u]
#define pp par[uu]

void Make(int x){R[x] = 0, P[x] = x, A[x] = x;}
int Find(int x){return x == P[x] ? x : P[x] = Find(P[x]);}
void Union(int _x, int _y){
int x = Find(_x), y = Find(_y);
if (R[x] < R[y]) P[y] = x;
else{
P[x] = y, A[y] = _x;
if (R[x] == R[y]) ++R[x];
}
}

inline int new_node(){
RST(trans[tot]); //cnt[tot] = 0;
return tot++;
}

inline int new_node(int u){
CPY(trans[tot], trans[u]); par[tot] = par[u]; //cnt[tot] = cnt[u];
return tot++;
}

inline int h(int u){
return len[u] - len[p];
}

int Ext(int c, int tail){
int u = tail, uu = new_node(); len[uu] = len[u] + 1;
while (u && !v) v = uu, u = p; // 向上遍历没有 c-转移 的祖先 ..
if (!u && !v) v = uu, pp = 0;
else{
if (len[v] == len[u] + 1) pp = v;
else{
int _v = v, vv = new_node(_v); len[vv] = len[u] + 1; par[_v] = pp = vv;
while (u && v == _v) v = vv, u = p;
if (!u && v == _v) v = vv;
}
}
return uu;
}

#undef v
#define v (*it)

void tarjan(int u = 0){

Make(u); ECH(it, adj[u]) tarjan(v), Union(u, v);

if(MtoT[u]) ECH(it, Trie::b[MtoT[u]]){
--cnt[A[Find(L[v])]]; ++cnt[u], L[v] = u;
}
}

void dfs1(int u = 0){
ECH(it, adj[u]) dfs1(v), cnt[u] += cnt[v];
}

void dfs2(int u = 0){
ECH(it, adj[u]) cnt[v] += cnt[u], dfs2(v);
ECH(it, Trie::b[MtoT[u]]) ans[v] += cnt[u];
}

#undef v
#define v Trie::trans[u][c]

void Init(){
tot = 0; queue<int> Q; Q.push(0); TtoM[0] = new_node();

while(SZ(Q)){
int u = Q.front(); Q.pop();
REP(c, 26) if (v) Q.push(MtoT[TtoM[v] = Ext(c, TtoM[u])] = v);
}

FOR(u, 1, tot) adj[p].PB(u); fill(L, L+nn, 0); tarjan();
dfs1(); FOR(u, 1, tot) cnt[u] = cnt[u] >= K ? h(u) : 0; dfs2();
}
}
