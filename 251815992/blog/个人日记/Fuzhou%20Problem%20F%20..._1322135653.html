void Make(int x){
P[x] = x, R[x] = 0;
}

int Find(int x){
if (P[x] != x) P[x] = Find(P[x]);
return P[x];
}

void Union(int x, int y){
if (R[x] < R[y]) P[y] = x;
else {
if (R[x] == R[y]) R[y]++;
P[x] = y;
}
}

void Kruskal(){
sort(E, E+m); REP(i, n) Make(i);

REP(i, n) CLR(adj[i]);

REP(i, m){
if (Find(E[i].a) == Find(E[i].b)) continue;
Union(Find(E[i].a), Find(E[i].b));
InMst[E[i].id] = true, adj[E[i].a].PB(MP(E[i].b, E[i].id)), adj[E[i].b].PB(MP(E[i].a, E[i].id));
mst += E[i].w;
}
}

void dfs(int u, int p){
REP(i, SZ(adj[u])){
int v = adj[u][i].first;
if (v != p) Fa[v] = u, Fb[v] = adj[u][i].second, dfs(v, u);
}
}

int lca(int a, int b){
while (a != 0){
vis[a] = _i;
a = Fa[a];
}

vis[0] = _i;

while (vis[b] != _i){
b = Fa[b];
}

return b;
}

void link(int a, int b, int c){

#define Cloze(a) if (!s[a]) s[a] = c

while (b != p){
Cloze(Fb[b]);
t = Fa[b], Fa[b] = p, b = t;
}
}
