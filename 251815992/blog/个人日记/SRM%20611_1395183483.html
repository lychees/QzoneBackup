一个 N 个结点的完全图的生成树方案由 n^(n-2) 种。。显然不能全部枚举。。。
分析法。。假设我们知道 均值 。。那么我们就可以构造新的图，求出原图的最小方差生成树。。。。
。。但是我们怎么可能在没有确定选定哪些边之前就准确的确定均值呢？。。


。。。我们把最小方差生成树看做  枚举的 拟均值 m 的函数。。。
。。显然这是个分段二次单谷函数。。。

。。。考察 Kruskal 算法的排序过程。。 Ei < Ej 。。有。。
sqr(wi -  m) < sqr(wj - m) ...
wi^2 -2mwi < wj^2 - 2mwj
2m(wi - wj) > wi^2 - wj^2 ...
m > wi^2 - wj^2 / 2(wi - wj)
m > (wi + wj) / 2 。。。
。。也就是说。。我们对所有的 (wi + wj) / 2 排序。。。
。。那么在任意相邻的两个值之间求出的 MST 方案都相同。。。
。。。这样还是枚举方案。。但是总的复杂度只有。。O(n4) 了。。

。。。最鬼畜的情况当属一开始的 卍 解。。。。。
。。。。。如果你按照图里面的那种全部 8 种都拿出来讨论那就直接跪了。。。
。。就算你能讨论清楚复杂度也不对了。。。
。。。所以最关键的部分是合并状态。。。以尽可能少的语言表达尽可能多的内容。。。。


我们设 dp x0 y0 x1 y1 表示状态，

。。那么大多数情况只能从 2 个方向延伸触手。。。实际上只有初始状态 有可能  需要用从四个方向延伸触手的状态去表达。。。
。。。。因此这个最外围的部分我们需要比较有技巧的枚举。。

。。。
考虑记忆化搜索。。。
int U=x0==0, D=x1==n, L=y0==0, R=y1==n;
。
分别表示能否从 上、下、左、右四个方向伸出触手。。。
。。

这里又可以分两类讨论。。

。。横竖都能延伸触手。。
。。和只能横或竖延伸触手。。两种。。

对于第一类转移时 O(n2)。。。。第二类转移 O(n)。。
。。总的状态只有 O(n2) 种。。（因为至少有一个方向要能延伸触手。。）

总的复杂度 O(n4)。。。。

const int N = 51;

int A[N][N], dp[N][N][N][N];
int n;

class ElephantDrinking {
public:

int f(int x0, int y0, int x1, int y1){
if (x0 >= x1 || y0 >= y1) return 0;
int& res = dp[x0][y0][x1][y1]; if (!~res){
res = 0; int U=x0==0, D=x1==n, L=y0==0, R=y1==n;
if (!L && !R){
int m = 0, mm = 0; FOR(x, x0, x1){
if (A[x][y0]>m)mm=m,m=A[x][y0];
else checkMax(mm, A[x][y0]);
}
res = (U+D==2?m+mm:m) + f(x0, y0+1, x1, y1);
}
else if (!U && !D){
int m = 0, mm = 0; FOR(y, y0, y1){
if (A[x0][y]>m)mm=m,m=A[x0][y];
else checkMax(mm, A[x0][y]);
}
res = (L+R==2?m+mm:m) + f(x0+1, y0, x1, y1);
}
else{
if (U) FOR(x, x0, x1) FOR(y, y0, y1) checkMax(res, A[x][y]+f(x0,y0,x+1,y)+f(x0,y+1,x+1,y1)+f(x+1,y0,x1,y1));
if (D) FOR(x, x0, x1) FOR(y, y0, y1) checkMax(res, A[x][y]+f(x,y0,x1,y)+f(x,y+1,x1,y1)+f(x0,y0,x,y1));
if (L) FOR(x, x0, x1) FOR(y, y0, y1) checkMax(res, A[x][y]+f(x0,y0,x,y+1)+f(x+1,y0,x1,y+1)+f(x0,y+1,x1,y1));
if (R) FOR(x, x0, x1) FOR(y, y0, y1) checkMax(res, A[x][y]+f(x0,y,x,y1)+f(x+1,y,x1,y1)+f(x0,y0,x1,y));
}
}
return res;
}


int maxSum(vector<string> field){
n = SZ(field); REP_2(i, j, n, n) A[i][j] = field[i][j] - '0';

FLC(dp, -1); int res = f(0,0,n,n); REP_1(a, n) FOR_1(b, n-a+1, n) REP_1(c, n) FOR_1(d, n-c+1, n)
checkMax(res, max(f(0,0,n-b,c)+f(n-b,0,n,n-d)+f(a,n-d,n,n)+f(0,c,a,n),
f(0,0,a,n-d)+f(a,0,n,c)+f(n-b,c,n,n)+f(0,n-d,n-b,n)));
return res;
}
};
