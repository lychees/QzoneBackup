算法 1 ：。。
预处理出关键点 + 单调队列 DP。。。O(n3)



const int N = 110;
LL dp[2][2*N*N], h[N], n, d; vector<LL> a;

int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

Rush{

RD(n, d); REP(i, n) RD(h[i]);

if (abs(h[0]-h[n-1])>d*(n-1)){
puts("impossible");
continue;
}

CLR(a); REP(i, n) FOR_1(j, -(n-1), n-1) a.PB(h[i]+d*j); UNQ(a);
int p = 0, q = 1; FLC(dp[p], 0x3f); dp[p][LBD(a, h[0])] = 0;

FOR(i, 1, n){
swap(p, q); FLC(dp[p], 0x3f); int k=0; REP(j, SZ(a)){
while (a[k]<a[j]-d || dp[q][k+1]<=dp[q][k] && a[k+1]<=a[j]+d) ++k;
if (dp[q][k] != INFF) dp[p][j] = dp[q][k] + abs(h[i]-a[j]);
}
}

cout << dp[p][LBD(a, h[n-1])] << endl;
}
}

const int N = int(6e3) + 9;
LL a[N], b[N]; LL n, d;
int main(){
#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif
Rush{
RD(n, d); REP(i, n) RD(a[i]);
if(abs(a[n-1]-a[0])>(n-1)*d){
puts("impossible");
continue;
}
b[0] = a[0]; FOR(i, 1, n) b[i] = b[i-1]-d;
while (a[n-1] != b[n-1]){
int p=0,pp=-INF,s; LL d=INFF,dd; DWN(i, n, 1){
if (b[i]<a[i])++p,checkMin(d, a[i]-b[i]);else--p;LL t = b[i-1]+::d-b[i];
if (p > pp && t) dd = min(d, t), pp = p, s = i;
}
FOR(i, s, n) b[i] += dd;
}
LL ans = 0; FOR(i, 1, n) ans += abs(a[i]-b[i]);
cout << ans << endl;
}
}
