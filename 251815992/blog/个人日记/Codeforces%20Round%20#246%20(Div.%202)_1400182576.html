给定一个字符串。。。问哪些前缀也是后缀。。
并且计数这些前缀在字符串中的出现次数。。。



我的做法是后缀数组。。。。
枚举合法后缀，二分出其在后缀数组中，向右延伸的位置。。。
以判断其是否覆盖前缀、并 求出其在字符串中出现的次数 。 。。。。

VII res; FOR_1(i, 1, rk[0]){
int x = i, l = x, r = n;
int len = n-sa[x];

while (l < r){
int m = l + r + 1 >> 1;
if (lcp(x, m) < len) r = m - 1;
else l = m;
}

if (x <= rk[0] && rk[0] <= l) res.PB(MP(len, l-x+1));
}

namespace KMP{

void get_pi(const char P[], int n, int pi[]){
for (int i = 1, j = pi[0] = -1; i < n; ++i){
while (j >= 0 && P[i] != P[j+1]) j = pi[j];
if (P[i] == P[j+1]) ++j;
pi[i] = j;
}
//REP(i, n) cout << pi[i] << " "; cout << endl;
}

int run(const char T[], int n, const char P[], int m, const int pi[]){
int cnt = 0; for (int i = 0, j = -1; i < n; ++i){
while (j >= 0 && T[i] != P[j+1]) j = pi[j];
if (T[i] == P[j+1]) ++j;
if (j == m - 1) ++cnt, j = pi[j];
}
return cnt;
}
} using namespace KMP;

const int N = int(1e5) + 9;
char str[N]; int pi[N], dp[N]; bool ck[N]; int n;


int main(){

#ifndef ONLINE_JUDGE

int i = n-1, z = 0; while (~i) ++z, ck[i] = 1, i = pi[i]; OT(z);
dp[n-1] = 1; REP(i, n) ++dp[pi[i]]; DWN(i, n, 0) dp[pi[i]] += dp[i];
REP(i, n) if (ck[i]) printf("%d %d\n", i+1, dp[i]);
}
