const int PMAX = 46341;
VI P; bitset<PMAX> isC;
#define ii (i*P[j])
void sieve(){
FOR(i, 2, PMAX){
if (!isC[i]) P.PB(i);
for (int j=0;j<SZ(P)&&ii<PMAX;++j){
isC[ii]=1; if (!(i%P[j])) break;
}
}
}
#undef ii

const int N = 1009;
LL f[2][N]; int n;

int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

sieve(); RD(n); int p = 0, q = 1; fill(f[p], f[p]+n+1, 1);

for (int i=0;i<SZ(P)&&P[i]<=n;++i){
swap(p, q); CPY(f[p], f[q]);
for (int ii=P[i];ii<=n;ii*=P[i]){
DWN_1(j, n, ii) f[p][j] += f[q][j-ii];
}
}

cout << f[p][n] << endl;
}
