const int N = int(2e6) + 9, K = int(1e6) + 9;

int n, m, k, s0, s1, s11;

namespace DSU{ // Disjoint Set Union
int P[N], F[N], rk;
inline void Make(int x){
P[x] = x, F[x] = 0;
}
inline int Find(int x){
//return P[x] == x ? x : P[x] = Find(P[x]);
if (P[x] == x) return x;
int t = Find(P[x]); F[x] ^= F[P[x]];
return P[x] = t;
}
inline void Unionn(int x, int y, int f){
//if (R[x] == R[y]) ++R[x];
//else if (R[x] < R[y]) swap(x, y);
}
inline bool Union(int x, int y, int f){
int xx = Find(x), yy = Find(y);

if (xx == yy){
return F[x]^F[y] == f;
}
else{
P[xx] = yy, F[xx] = F[x]^F[y]^f;
--rk;
return 1;
}

}
inline void Init(int n){
REP(i, n+1) Make(i);
}
} using namespace DSU;

struct edge{
int r,c,f;
void in(){
RD(r,c,f);
if (r==1&&c==1) s11 = f;
}

bool process(){
if (r==1){
return Union(c+n,1,f^s11);
}
else if (c==1){
return Union(r,1,f^s11);
}
else{
int ff = !(r&1)&&!(c&1); ff ^= s11;
return Union(r,c+n,f^ff);
}
}

} E[K];

bool ok(){
REP(i, k) if (!E[i].process()) return 0;
return 1;
}
