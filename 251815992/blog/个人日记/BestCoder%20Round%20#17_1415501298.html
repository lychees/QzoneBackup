好题。。。

O(n3n)  算法：
f[ s][u]：表示以 u 为根 合法的有根树方案数。。
g[s]：合法的有根树方案数。。。（以任意 s 中的结点为根）。。

转移就是枚举子集。。。

f[s][i] = f[][i] * g[ss];
ss 不能包含 i。。并且为了避免重复计数。。还必须固定一个 i 以外的一个结点给 ss。。。





（子集 DP？。。参见 http://user.qzone.qq.com/251815992/blog/1414062562 ）


//}/* .................................................................................................................................. */

const int N = 14;
int l[N], r[N], n;
Int f[1<<N][N], g[1<<N];

int main() {

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

Rush{
REP_C(i, RD(n)) RD(l[i],r[i]);

RST(f, g); g[0] = 1; REP(i, n) f[_1(i)][i] = 1;

int S = _U(n); REP_1(s, S) REP(i, n) if (_1(s, i)){
int x = low_idx(s); if (x == i) x = high_idx(s);
#define ss (_ss|_1(x))
int SS = s^_1(i)^_1(x);
REP_SS(_ss, SS) f[s][i] += f[s^ss][i] * g[ss|_1(x)];
int _ss = 0; f[s][i] += f[s^ss][i] * g[ss|_1(x)];

if (l[i] <= count_bits(s) && count_bits(s) <= r[i]) g[s] += f[s][i];
}

printf("%d\n", int(g[S]));
}
}
