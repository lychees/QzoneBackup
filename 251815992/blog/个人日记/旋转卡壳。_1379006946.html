int n=SZ(P); if(n<=3) return P.PB(P[0]),getArea(P)<0?RVS(P):P;

SRT(P); VP C; C.resize(n+9); int nn = -1; REP(i, n){ //#
while (nn > 0 && dett(C[nn-1], C[nn], P[i]) <= 0) --nn; //#
C[++nn] = P[i];
}

int _nn = nn; DWN(i, n-1, 0){
while (nn > _nn && dett(C[nn-1], C[nn], P[i]) <= 0) --nn; //#
C[++nn] = P[i];
}

C.resize(nn+1);
return C;
}

#define suc(x, n) (x+1==n?0:x+1)
DB _rc(const VP& P1, const VP& P2){
int n = SZ(P1)-1, m = SZ(P2)-1;
int i=0, j=0; DB d2=OO;

REP(k, n) if (P1[k].y > P1[i].y) i = k;  //#
REP(k, m) if (P2[k].y < P2[i].y) i = k;

DO(n){
Seg h(P1[i], P1[i+1]); while (dett(h.d(), P2[j+1]-P2[j])>0) j=suc(j,m);
checkMin(d2, dist2(h, Seg(P2[j],P2[j+1])));
i=suc(i,n);
}
return d2;
}
DB rc(const VP& P1, const VP& P2){
return min(_rc(P1, P2), _rc(P2, P1));
}

。。凸包内面积最大三角。（复杂度整个增加一维。。。似乎有优化的余地。。
#define suc(x) (x+1==n?0:x+1)
DB rc(const VP& P){
int n = SZ(P)-1; DB res = 0; int j, k; REP(i, n){
for (j=k=suc(i);j!=i;j=suc(j)){
while (dett(P[j]-P[i], P[k+1]-P[k])>0) k=suc(k);
checkMax(res, fabs(det(P[i], P[j], P[k])));
}
}
return res/2;
}
