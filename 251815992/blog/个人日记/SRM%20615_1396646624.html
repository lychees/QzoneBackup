无向图。。问 s->t 是否存在长度为 T 的路径。。。

。。。。最终的结果可能在图中绕来绕去。。。如何避免考虑这种复杂情况呢？
。。观察某条从 s 出发的边。。。假设其边长为 m。。那么如果 D 可行。。那么任意的 D+2km | k >= 0 都可行。。。
。。也就是说。。只要对 T % 2km 求出最小值即可。。。而这相当于在模 2km 意义下求最短路。。。。

。。。不难发现。。任意选一条 s 出发的边都可以。。。我们选边长最小的。。。
复杂度是 O( 最短路*m) 。。

const int N = 50, M = 20000;
VII adj[N]; LL D[N][M];
int n, m;


class LongLongTripDiv1 {
public:
string isAble(int n, vector <int> A, vector <int> B, vector <int> DD, long long T) {

REP(i, n) CLR(adj[i]); REP(i, SZ(A)){
adj[A[i]].PB(MP(B[i], DD[i]));
adj[B[i]].PB(MP(A[i], DD[i]));
}

if (adj[0].empty()) return "Impossible";

int m = adj[0][0].se; FOR(i, 1, SZ(adj[0])) checkMin(m, adj[0][i].se); m *= 2;
FLC(D, 0x3f); D[0][0] = 0; priority_queue<PII, VII, greater<PII> > Q; Q.push(MP(0, 0));

while (!Q.empty()){
int u = Q.top().se, d = Q.top().fi; Q.pop();
if (d != D[u][d%m]) continue; ECH(it, adj[u]){
int v = it->fi, w = it->se;
if (d+w<D[v][(d+w)%m]){
D[v][(d+w)%m] = d+w;
Q.push(MP(d+w, v));
}
}
}

return D[n-1][T%m] <= T ? "Possible" : "Impossible";
}
};

dp[i][j][k]:
到第 i 个字符，当前 （ 括号多出来 j 个。。。已经产生了 k 个匹配。。。

之所以要第三维是因为答案要求产生的总的匹配数是 m 的倍数。。。因此最后一维可以对 m 取模。。。
。。。实际上后面两维都开到 m 就可以了。。状态总数是 O(nmm) 转移 O(1)。。。
