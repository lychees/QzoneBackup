1000：

。给定一个无根树(n = 200）。。。选择一些叶子。。同时点燃。。
。问烧光整个树所需要的时间有多少种可能。。。？


1.
方案不会很多。
—— 只可能是 D[u][v] 或者 D[u][v] / 2。









int N;
vector<pint> G[1010];
int D[1010][1010];

int dists[1010];

bool check(int a, int da, int b, int db, int l) {
// cout<<"check "<<a<<" "<<da<<"  "<<b<<" "<<db<<"  "<<l<<endl;
priority_queue<pint,vector<pint>,greater<pint>> q;
fill(dists, dists + N, INF);
for (int u = 0; u < N; ++u) if (G[u].size() == 1) {
if (D[u][a] <= D[u][b]) {
if (D[u][a] + da >= l) {
dists[u] = 0;
q.emplace(0, u);
}
} else {
if (D[u][b] + db >= l) {
dists[u] = 0;
q.emplace(0, u);
}
}
}
for (; !q.empty(); ) {
const int c = q.top().first;
const int u = q.top().second;
q.pop();
if (dists[u] == c) {
for (const auto &e : G[u]) {
const int cc = c + e.first;
const int v = e.second;
if (dists[v] > cc) {
dists[v] = cc;
q.emplace(cc, v);
}
}
}
}
for (int u = 0; u < N; ++u) {
for (const auto &e : G[u]) {
const int v = e.second;
if ((dists[u] + dists[v] + e.first) / 2 > l) {
return false;
}
}
}
return true;
}

struct CandleTimer {
int differentTime(vector <int> A, vector <int> B, vector <int> len) {
N = A.size() + 1;
for (int u = 0; u < N; ++u) {
fill(D[u], D[u] + N, INF);
D[u][u] = 0;
}
for (int i = 0; i < N - 1; ++i) {
const int u = A[i];
const int v = B[i];
const int c = len[i] * 2;
G[u].emplace_back(c, v);
G[v].emplace_back(c, u);
chmin(D[u][v], c);
chmin(D[v][u], c);
}
for (int w = 0; w < N; ++w) for (int u = 0; u < N; ++u) for (int v = 0; v < N; ++v) {
chmin(D[u][v], D[u][w] + D[w][v]);
}

set<int> ans;
for (int u = 0; u < N; ++u) {
for (int w = 0; w < N; ++w) if (w != u) {
if (ans.count(D[u][w])) {
continue;
}
if (check(w, 0, G[w][0].second, G[w][0].first, D[u][w])) {
ans.insert(D[u][w]);
}
}
}
for (int u = 0; u < N; ++u) for (int v = u + 1; v < N; ++v) if (G[u].size() == 1 && G[v].size() == 1) {
if (ans.count(D[u][v] / 2)) {
continue;
}
// cout<<"midpoint "<<u<<" "<<v<<endl;
for (int w = 0; w < N; ++w) {
for (const auto &e : G[w]) {
const int x = e.second;
if (D[u][v] == D[u][w] + e.first + D[x][v]) {
if (D[u][w] <= D[u][v] / 2 && D[u][v] / 2 < D[u][w] + e.first) {
if (check(w, D[u][v] / 2 - D[u][w], x, D[u][w] + e.first - D[u][v] / 2, D[u][v] / 2)) {
ans.insert(D[u][v] / 2);
}
goto found;
}
}
}
}
assert(false);
found:;
}
cout<<"ans : ";pv(ans.begin(),ans.end());
return ans.size();
}
};
