int i, j, k, l; int *X = t1, *Y = t2;

for (i=0;i<n;i++) A[i] = s[i];
m = M;
memset(C, 0, sizeof(C));
for (i=0;i<n;i++) C[X[i] = A[i]]++;
for (i=1;i<m;i++) C[i] += C[i-1];
for (i=n-1;i>=0;i--) SA[--C[X[i]]] = i;
X[n] = Y[n] = INF;
for (l=1;l<n;l<<=1){
for (i=n-l,j=0;i<n;i++) Y[j++] = i;

for (i=0;i<n;i++) key[i] = X[Y[i]];
memset(C, 0, sizeof(C));
for (i=0;i<n;i++) C[key[i]]++;
for (i=1;i<m;i++) C[i] += C[i-1];
for (i=n-1;i>=0;i--) SA[--C[key[i]]] = Y[i];
swap(X, Y);

j++;
X[SA[i]] = j;
}

if (m==n) break;
}
for (i=0;i<n;i++)
RA[SA[i]] = i;
k = 0;
for (i=0;i<n;i++){
if (k>0) k--;
if (RA[i]==0) height[0] = 0;
else {
j = SA[RA[i] - 1];
while (A[i+k]==A[j+k]) k++;
height[RA[i]] = k;
}
}
}

int length = 1, pos = 0;
/*for (int i=0;i<n;i++)

cout << endl; */
for (int i=0;i<n-1;i++){
if (height[i+1]>=length) {
for (int j=i+1;j<n;j++){
if ((SA[i] + SA[j] + height[i+1] == n)){
if (height[i+1] > length) length = height[i+1], pos = min(SA[i], SA[j]);
