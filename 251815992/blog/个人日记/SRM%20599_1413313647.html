static const int MAX_VALID_MASKS = 6561;
const int N = 51, M = 16; VI adj[N]; int n, m, sn;

int validSets[MAX_VALID_MASKS]; int id[1<<M];
VI validSubsets[MAX_VALID_MASKS];
Int f[N][MAX_VALID_MASKS];

class SimilarNames {
public:

void dfs(int u) {

REP(i, sn) f[u][i] = 0; f[u][0] = 1; ECH(it, adj[u]){
int v = *it;
dfs(v); DWN(i, sn, 0){
Int t = 0; int s1 = validSets[i];
for (int s2: validSubsets[i]) t += f[v][id[s1^s2]] * f[u][id[s2]];
f[u][i] = t;
}
}

if (!u) return;

DWN(i, sn, 0){
int s1 = validSets[i];
Int t = f[u][i]; REP(j, m) if (_1(s1, j) && ~id[s1^_1(j)]) t += f[u][id[s1^_1(j)]];
f[u][i] = t;
}
}

bool isPrefix(string a, string b){
return a.length() < b.length() && b.substr(0, a.length()) == a;
}

int count(vector <string> names, vector <int> info1, vector <int> info2){

names.PB(""); SRT(names); n = names.size();

REP(i, n) adj[i].clear(); FOR(i, 1, n){
int p = 0; FOR(j, 1, i) if (isPrefix(names[j], names[i]) && names[j].length() > names[p].length()) p = j;
adj[p].PB(i);
}

MII H; m = 0;

ECH(it, info1){
if (!CTN(H, *it)) H[*it] = m++;
*it = H[*it];
}
ECH(it, info2){
if (!CTN(H, *it)) H[*it] = m++;
*it = H[*it];
}

FLC(id, -1); sn = 0; REP(s, _1(m)){
bool ok = 1; REP(i, SZ(info1)) if (_1(s, info1[i]) && !_1(s, info2[i])){
ok = 0;
break;
}

if (ok){
validSets[sn] = s;
validSubsets[sn].clear();
id[s] = sn++;
}
}

REP_2(i, j, sn, sn){
if ( (validSets[i] & validSets[j]) == validSets[j] ) {
if ( id[ validSets[i] ^ validSets[j] ] != -1 ) {
validSubsets[i].PB( validSets[j]);
}
}
}

RST(f); dfs(0); Int ans = f[0][sn-1]; REP_1(i, n-1-m) ans *= i;
return (int)ans;
}
};
