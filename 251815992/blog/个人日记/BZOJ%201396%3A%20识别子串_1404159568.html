对于任意一个询问 x。。。
。。我们讨论【极小识别字串】是否包含 x 。。然后两者取最小。
（如果包含。。那么就取长度最小。。。如果不包含。。那么取右端点最小。）
对于这个 Sub Problem。。。。我们线段树。。或者离线乱搞怎么做都好。。



——————————————
。。因此我们主要的任务也就是求出一个字符串的所有【极小识别子串】（不包含其它更小的识别子串）。。
而这个利用 SAM 的 par 树可以方便求出。。。。。

const int N = int(1e5) + 9, NN = 2*N, Z = 26;

namespace SAM{
int trans[NN][Z], par[NN], len[NN], tail, tot;
bool bj[NN]; char buf[N]; int n;

int new_node(){
//RST(trans[tot]);
return tot++;
}

int new_node(int u){
CPY(trans[tot], trans[u]), par[tot] = par[u];
return tot++;
}

#define v trans[u][c]
#define p par[u]
#define pp par[uu]

void Ext(int c){
int u = tail, uu = new_node(); len[uu] = len[u] + 1;
while (u && !v) v = uu, u = p;
if (!u && !v) v = uu, pp = 0;
else{
if (len[v] == len[u] + 1) pp = v;
else{
int _v = v, vv = new_node(_v); len[vv] = len[u] + 1;
par[_v] = pp = vv;
while (v == _v) v = vv, u = p;
}
}
tail = uu;
}

void Init(){
tail = 0, tot = 1; n = strlen(RS(buf)); REP_S(cur, buf) Ext(*cur - 'a');
FOR(u, 1, tot) bj[p] = 1;
}

//#undef v
//#undef p

} using namespace SAM;

struct core{
int l, r;
core(int l, int r):l(l),r(r){}
inline int len()const{return r-l+1;}
};

bool cmp_l(const core&a, const core&b){
return a.l<b.l;
}
bool cmp_r(const core&a, const core&b){
return a.r<b.r;
}

struct cmp_len{
bool operator()(const core&a, const core& b){
return a.len() >= b.len();
}
};

priority_queue<core, vector<core>, cmp_len> Q; int min_r;
vector<core> Il, Ir; int ans[N];

int main(){

}

SRT(Il, cmp_l); SRT(Ir, cmp_r);
Q.push(core(0, INF)), min_r = INF;

int ir = SZ(Ir)-1, il = ir; DWN_1(i, n, 1){
if (ir >= 0 && Ir[ir].r == i) Q.push(Ir[ir--]);

while (Q.top().l > i) Q.pop();
ans[i] = min(min_r-i+1, Q.top().len());
}

REP_1(i, n) OT(ans[i]);
