const int N = int(1e5) + 9;
const int PMAX = N;
VI P; bitset<PMAX> isP; int mu[PMAX];

void sieve(){
mu[1] = 1; FOR(i, 2, PMAX){
if (!isP[i]) P.PB(i), mu[i] = -1;
for (int j=0;j<SZ(P)&&i*P[j]<PMAX;++j){
isP[i*P[j]]=1; if (!(i%P[j])){
mu[i*P[j]] = 0;
break;
} else{
mu[i*P[j]] = -mu[i];
}
}
}
}
Int Fact[N], Factt[N]; Int Binom(int n, int m){
return Fact[n] / Fact[m] / Fact[n-m];
}

Int f(int n, int m){
if (n < m) return 0;
n -= m; --m; n += m;
return Binom(n, m);
}
