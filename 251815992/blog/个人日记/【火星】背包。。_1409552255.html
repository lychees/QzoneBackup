else if (n*c >= m){
REP(i, m-c+1) checkMax(dp[i+c], dp[i]+v);

else{
//VI I; int x=n, b=0; for (;x>=_1(b+1);++b) I.PB(_1(b)); I.PB(x-_1(b)+1);

CPY(dpp, dp); for (int r=0;r<c;++r){
