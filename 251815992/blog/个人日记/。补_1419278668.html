构造：

按照时间排序。

如果最慢的完成时间超过了其余的和。。那么答案显然是最慢的 * 2
否则。。

在最慢的完成一门课之后。总是可以让其去听第二门课。。
并且此时。。第一门课会 free 下来。。让其他课按照任意次序无缝完成第一门课。。。

const int N = int(1e3) + 9;

struct edge{
int to, ef; bool super;
void in(){
RD(to, ef, super);
}
}; vector<edge> adj[N];

int need[N]; DB dp[N];
int n;

#define v it->to

void dfs(int u, int p = -1){
ECH(it, adj[u]) if (v != p){
dfs(v, p);
if (it->super){
checkMax(dp[u], sqrt(dp[v]) * 100. / it->ef);
}
else{
checkMax(dp[u], dp[v] * 100. / it->ef);
}
}

checkMax(dp[u], (DB)need[u]);
}

int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

DO(RD(n)-1){
int u; RD(u); edge t; t.in();
adj[u].PB(t);
}

REP_1(i, n) RDD(need[i]);

dfs(1); OT(dp[1]);
}
