SCDP？？？

给定 n*m 的矩阵，每个格子有 E 或 S。。。
对于一个矩阵。。有一个计分方案。。按照行优先的规则扫描每个格子， 如果这个格子之前被骨牌占据，则 skip。
否则 尝试放多米诺骨牌。 如果放骨牌的方向在矩阵外或被其他骨牌占据，则放置失败，切换另一种方案或 skip。
如果是 E 则优先放一个 1*2 的骨牌，
如果是 S 则优先放一个 2*1 的骨牌。
一个矩阵的得分为最后放的骨牌数。
问所有 2^(n*m) 种矩阵的得分的和。

const int N = 14 ;


PII f[ 2 ][ 1 <<N]; int src, des;

// ways, score


PII u; int s, ss, i, j, W, H;


void skip(){

PII& v = f[des][s];

INC(v.fi, u.fi);

INC(v.se, u.se);

}

void put(){

PII& v = f[des][ss];

INC(v.fi, u.fi);

INC(v.se, sum(u.fi ,u.se));

}


bool tryPutDown(){

if (i == H- 1 ) return false ;

ss = s ^ _1(i+ 1 ); put();

return true ;

}


bool tryPutRight(){

if (j == W- 1 || _1(s, i)) return false ;

ss = s ^ _1(i); put();

return true ;

}


class BearDestroys {

public :

int sumUp( int W, int H, int MOD) {

::W = W, ::H = H; ::MOD = MOD;

src = 0 , des = 1 ; RST(f[des]); f[des][ 0 ] = MP( 1 , 0 );

REP(di, W+H- 1 ){ // 对角线

int ia = max( 0 , di-W+ 1 ), ib = min(H, di+ 1 );

FOR_N(i, ia, ib){

j = di-i; swap(src, des); RST(f[des]);

REP_N(s, 1 <<(H+ 1 )) if (f[src][s] != MP( 0 , 0 )){

u = f[src][s];

// Coverd

if (_1(s, i+ 1 )){

s ^= _1(i+ 1 );

DO( 2 ) skip();

s ^= _1(i+ 1 );

}

else {

if (!tryPutRight())

if (!tryPutDown()) skip();

if (!tryPutDown())

if (!tryPutRight()) skip();

}

}

}

swap(src, des); RST(f[des]);

REP_N(s, 1 <<H) if (f[src][s] != MP( 0 , 0 )){

f[des][s<< 1 ] = f[src][s];

}

}

return f[des][ 0 ].se;

}
