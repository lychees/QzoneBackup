inline int gcd(int m, int n, int &x, int &y){

}

inline int _I(int b){

const int N = 1<<16;
int a[N];

void FWT(){
static LL aa[N]; REP(i, N) aa[i] = a[i];
for (int i=1;i<N;i<<=1){
REP(j, N) if (~j&i){
aa[j] += aa[j|i];
aa[j|i] = aa[j]-(aa[j|i]<<1);
}
}
REP(i, N){
a[i] = aa[i] % MOD;
if (a[i] < 0) a[i] += MOD;
}
}

bool isP(int x){
if (x <= 1) return 0;
for (int i=2;i*i<=x;++i)
if (x%i==0) return 0;
return 1;
}

}
};
