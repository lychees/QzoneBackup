Problem E.

。。设 n 表示数个数。。m 表示区间数。。。

n*m <= 1e5，且 n < m 时无解。
因此数组只需开到 sqrt(1e5) 即可。

dp[ i ][ j ][ k ] ：
表示前 i 个数，开启了 j 个区间，其中 k 个没有封闭的方案数。

转移分：
#define u f[q][j][k]
f[p][j+1][k+1] += u;  // 新开一个区间。
f[p][j+1][k] += u; // 新开一个区间并立刻封闭。
f[p][j][k] += u; // 跳过这个数。
f[p][j][k-1] += u; // 封闭一个区间
四中情况讨论即可。。其中后两种在 x 位置时失效。


const int N = int(3.4e2) + 9;
Int f[2][N][N];
int n, m, x;
int solve(){
if (m > n) return 0;
int p = 0, q = 1; f[p][0][0] = 1; REP_1(i, n){
swap(p, q); REP(j, min(i, m)+1) REP(k, j+1) f[p][j][k] = 0; REP(j, min(i, m+1)) REP(k, j+1){
#define u f[q][j][k]
f[p][j+1][k+1] += u;
f[p][j+1][k] += u;
if (i != x){
f[p][j][k] += u;
if (k) f[p][j][k-1] += u;
}
}
}
Int res = f[p][m][0]; REP_1(i, m) res *= i;
return res;
}

int main(){
#ifndef ONLINE_JUDGE
freopen("in.txt","r",stdin);
//freopen("output.txt","w",stdout);
#endif
RD(m, n, x);
OT(solve());
}
