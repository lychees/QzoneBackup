Brief description:

给定一组点集（第一象限），每次询问一个三角形。（其中一个顶点在原点）
问在该三角形的内部是否存在点集中的点。。。

——————

Solution:

首先考虑所有三角形都是一条边在 x 轴上，一条边在 y 轴的直角三角形。。。
那么问题等于。。测试一条直线。是否与一段凸壳相交。。。
。。。直线与凸壳的距离单峰。。。所以可以二分答案。


对于一般情况。。。就是测试。。在某段夹角内的点集的凸壳。。与该直线是否与一条直线相交。。。
。。。可以对点集按照极角排序。。然后用线段树分割区间。。。每次在线段树中找出对应的区间。。分别测试即可。。

预处理：O(nlogn + nlog^2n )
每次询问复杂度：O(log^2n )

#define cPo const Po&

struct Po{
int x, y; inline Po(int x=0, int y=0):x(x),y(y){} void in(){RD(x, y);}
Po operator -(cPo r)const{return Po(x-r.x,y-r.y);}
LL operator *(cPo r)const{return (LL)x*r.y-(LL)y*r.x;}
bool operator <(cPo r) const {return Ts*r>0;}
};

const int N = 100001;
Po P[N]; vector<Po> C[4*N]; int n, Q;

#define lx (x<<1)
#define rx (lx|1)
#define ml (l+r>>1)
#define mr (ml+1)
#define lc lx,l,ml
#define rc rx,mr,r
#define root 1,0,n-1

#define C C[x]

void Build(int x, int l, int r){
if(l==r) C.PB(P[l]);
else{
C.resize(r-l+1), C[0] = P[r], C[1] = P[r-1]; int nn=1;
for(int i=r-2;i>=l;i--){
while(nn>0&&(C[nn]-C[nn-1])*(P[i]-C[nn-1])<=0)--nn;
C[++nn]=P[i];
}
C.resize(nn+1);
Build(lc),Build(rc);
}
}

Po pa, pb; int a, b;

inline bool ck(cPo x){
return pa*x+x*pb<=pa*pb;
}

bool Query(int x,int l,int r){
if(a<=l&&r<=b){
Po v = pb-pa; l=0,r=SZ(C)-1; while(l<r){
if(v*(C[mr]-C[ml])>0)l=mr;
else r=ml;
}
return ck(C[l]);
}

return a < mr && Query(lc) || ml < b && Query(rc);
}
int main()
{

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

RD(n, Q); REP(i, n) P[i].in(); sort(P, P+n);

Build(root); DO(Q){
pa.in(), pb.in(); if(pb<pa)swap(pa,pb);
a = lower_bound(P,P+n,pa)-P, b = lower_bound(P,P+n,pb)-P-1;
puts(Query(root) ? "Y" : "N");
}
return 0;
}

Po P[N]; VI C;
int n, Q;

#define lx (x<<1)
#define rx (lx|1)
#define ml (l+r>>1)
#define mr (ml+1)
#define lc lx,l,ml
#define rc rx,mr,r
#define root 1,0,n-1

Po pa, pb; int a, b;

inline bool ck(cPo x){
return pa*x+x*pb<=pa*pb;
}

int main()
{

//freopen("out.txt", "w", stdout);

C.resize(n), C[0]=n-1, C[1]=n-2; int nn=1; for(int i=n-3;i>=0;i--){
while(nn>0&&(P[C[nn]]-P[C[nn-1]])*(P[i]-P[C[nn-1]])<=0)--nn;
C[++nn]=i;
}
C.resize(++nn); RVS(C);

DO(Q){
pa.in(), pb.in(); if(pb<pa)swap(pa,pb);
a = lower_bound(P,P+n,pa)-P, b = lower_bound(P,P+n,pb)-P-1;
int l = upper_bound(ALL(C),a)-C.begin()-1, r = upper_bound(ALL(C),b)-C.begin();
Po v = pb-pa; while(l<r){
if(v*(P[C[mr]]-P[C[ml]])<0)l=mr;
else r=ml;
}

puts(ck(P[C[l]]) ? "Y" : "N");
}
return 0;
}
