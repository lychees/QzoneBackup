Brief description：
题目有点晦涩。。。

“It is enough to find the farthest from A of these points in which some crude oil（原油） occurs.”

x 轴上。。从 0 点向右的一段区域内都有原油。。。每次可以在某点向下打下一枚探针（borehole）。。。
、测试原油是否延伸到这个位置。。每个点打探针的代价不同。。。。问至少需要多少代价。。可以测试出原油的实际范围。。



Solution：
？？？

首先可以 O(n3) 暴力 DP。。。

const int N = 2009;
int dp[N][N];
int A[N], n;

int f(int l, int r){
int &res = dp[l][r]; if (res == INF){
FOR_1(i, l, r) checkMin(res, A[i] + max(f(l, i-1), f(i+1, r)));
}
return res;
}

int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

REP_C(i, RD(n)) RD(A[i]);
FLC(dp, 0x3f); REP(i, n) dp[i][i] = A[i], dp[i][i-1] = 0;
cout << f(0, n-1) << endl;
}


似乎显然是决策单调的（？）。

观察转移方程。。
FOR_1(i, l, r) checkMin(res, A[i] + max(f(l, i-1), f(i+1, r)));

。。高光： max(f(l, i-1), f(i+1, r))
我们知道 f(l, r) 是满足区间包含单调的。。。
因此随着 i 的增大。。。左边那项 f(l, i-1) 单调递增。
。。右边的 f(i+1, r) 则单调递减。。

那么我们可以二分找出满足。。
f(l, i-1) <= f(i+1, r) //
的最后一个 i。。。


那么 l 到 i 之间 max 函数取右半部。。i+1 到 r 之间取左半部。。。。
于是可以利用区间单调性进行剪枝。。

const int N = 2009;
int dp[N][N], A[N], n;

int f(int l, int r){
int &res = dp[l][r]; if (res == INF){

int ll = l, rr = r; while (ll < rr){
int mm = ll + rr + 1 >> 1;
if (f(l, mm-1) <= f(mm+1, r)) ll = mm;
else rr = mm-1;
}

rr = 0; DWN_1(i, ll, l) if (A[i] + rr < res) rr = f(i+1, r), checkMin(res, A[i]+rr);
rr = 0; FOR_1(i, ll+1, r) if (A[i] + rr < res) rr = f(l, i-1), checkMin(res, A[i]+rr);
}
return res;
}

int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

REP_C(i, RD(n)) RD(A[i]);
FLC(dp, 0x3f); REP(i, n) dp[i][i] = A[i], dp[i+1][i] = 0;
cout << f(0, n-1) << endl;
}
