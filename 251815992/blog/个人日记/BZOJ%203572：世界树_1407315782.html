const int N = int(3e5)+9, M=2*N, LV = 19;

int hd[N],to[M],suc[M];
int n,m,fa[LV+1][N],dfn[N],dep[N],sz[N],tt;

inline bool cmp(const int lhs,const int rhs){return dfn[lhs] < dfn[rhs];}

#define aa to[i^1]
#define bb to[i]
#define v bb

void dfs(int u){
dfn[u] = ++tt, sz[u] = 1; REP_G(i, u) if (v != fa[0][u]){
dep[v] = dep[u]+1, fa[0][v] = u; for(int lv=0;fa[lv+1][v]=fa[lv][fa[lv][v]];++lv);
dfs(v), sz[u] += sz[v];
}
}

inline int move_up(int x, int t){
for (int lv=0;lv<LV&&t;++lv,t>>=1)
if (t&1) x = fa[lv][x];
return x;
}
inline int move_to(int x, int t){
return move_up(x, dep[x]-t);
}
inline int lca(int x, int y){
if (dep[x]>dep[y]) swap(x,y); y=move_to(y, dep[x]); if (x==y) return x;
DWN(lv, LV, 0) if (fa[lv][x]!=fa[lv][y]) x = fa[lv][x], y = fa[lv][y];
return fa[0][x];
}


void solve(){

static int fa[N],sta[N],rem[N],ans[N],top; top=0;
static PII g[N]; VI _T, T; Rush{
int x; RD(x); _T.PB(x); T.PB(x);
g[x] = MP(ans[x]=0, x);
}

#define w sta[top-1]
#define y sta[top]
#define x T[i]

SRT(T, cmp); REP_C(i, SZ(T)){
if (!top) fa[sta[++top] = x] = 0;
else {
int z = lca(x, y); for (; dep[y] > dep[z]; --top) if (dep[w] <= dep[z]) fa[y] = z;

if (y != z){
T.PB(z), g[z] = MP(INF,0);
fa[z] = y, sta[++top] = z;
}
fa[x] = z;
sta[++top] = x;
}
}

#undef x
#undef y
#undef w

#define u T[i]
#define p fa[u]
#define w (dep[u]-dep[p])

SRT(T, cmp); REP(i, SZ(T)) rem[u] = sz[u];
DWN(i, SZ(T), 1) checkMin(g[p], MP(g[u].fi+w,g[u].se));
FOR(i, 1, SZ(T)) checkMin(g[u], MP(g[p].fi+w,g[p].se));

REP(i, SZ(T)){
if (!i) ans[g[u].se] += n-sz[u];
else {
int pp = move_to(u, dep[p]+1), ss = sz[pp]-sz[u]; rem[p] -= sz[pp];
if (g[p].se == g[u].se) ans[g[u].se] += ss;
else {
int mid = dep[u] - ((g[p].fi + g[u].fi + w) / 2 - g[u].fi);
if ((g[p].fi + g[u].fi + w)%2==0 && g[u].se>g[p].se) ++mid;
int s = sz[move_to(u,mid)]-sz[u];
ans[g[u].se] += s, ans[g[p].se] += ss-s;
}
}
}

REP(i, SZ(T)) ans[g[u].se] += rem[u];
ECH(it, _T) printf("%d ", ans[*it]); puts("");
}

int main(){

#ifndef ONLINE_JUDGE

//freopen("out.txt", "w", stdout);
#endif

FOR_C(i, 2, RD(n)<<1){
RD(aa, bb);
suc[i] = hd[aa], hd[aa] = i++;
suc[i] = hd[aa], hd[aa] = i;
}

dfs(1); Rush solve();
}
