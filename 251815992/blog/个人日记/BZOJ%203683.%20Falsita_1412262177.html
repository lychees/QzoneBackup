const int N = 300009, M = 2 * N;
int A[N]; int L[N], R[N], up[N], fa[N], n, m, nn, rt;
int hd[N], suc[M], to[M];


LL sz[N], v0[N], sv[N], sz0[N], sz1[N], sz2[N];


#define aa to[i^1]
#define bb to[i]
#define v bb

inline void dfs(int u = 1){
sz[u] = 1; sv[u] = v0[u];

REP_G(i, u) if (v != fa[u]){
fa[v] = u; dfs(v), sz[u] += sz[v]; sv[u] += sv[v];
}

REP_G(i, u) if (v != fa[u]){
sz1[u] += sz[v] * (sz[u] - sz[v]);
sz2[u] += sv[v] * (sz[u] - sz[v]);
}

sz1[u] += sz[u] - 1;
sz2[u] += v0[u] * (sz[u] - 1);
}

inline void hld(int u = 1, int t = 1){
L[u] = ++nn, up[u] = t;
int h = 0; REP_G(i, u) if (v != fa[u] && sz[v] > sz[h])
h = v;

sz0[u] = sz[u] - sz[h];

if (h){
hld(h, t); REP_G(i, u) if (v != fa[u] && v != h)
hld(v, v);
}

R[u] = nn;
}


struct Tree{

LL T[4*N], d0[4*N], a, b; LL d;

#define rt 1, 1, n
#define lx (x<<1)
#define rx (lx|1)
#define ml (l + r >> 1)
#define mr (ml + 1)
#define xx x, l, r
#define lc lx, l, ml
#define rc rx, mr, r

inline void upd(int x, int l, int r){
T[x] = T[lx] + T[rx];
}

void inc(int x, int l, int r, LL d){
T[x] += d * (r-l+1);
d0[x] += d;
}

inline void rls(int x, int l, int r){
if (d0[x]){
inc(lx, l, ml, d0[x]);
inc(rx, mr, r, d0[x]);
d0[x] = 0;
}
}

void mdf(int x, int l, int r){
if (b < l || r < a) return;
if (a <= l && r <= b){
inc(xx, d);
return;
}

rls(xx);
mdf(lc),mdf(rc);
}

void Modify(int _a, int _b, LL _d){
a = _a, b = _b, d = _d;
mdf(rt);
}

LL Query(int x, int l, int r){
if (b < l || r < a) return 0;
if (a <= l && r <= b) return T[x]; rls(xx);
return Query(lc) + Query(rc);
}

LL Query(int _a, int _b){
//return 0;
a = _a, b = _b;
return Query(rt);
}

void Build(int x, int l, int r){
T[x] = d0[x] = 0;
if (l < r){
Build(lc), Build(rc);
}
}

} T_subtree, T_path;
