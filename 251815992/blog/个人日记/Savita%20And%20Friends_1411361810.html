const int N = int(1e5) + 9;

VI adj[N]; int n, m; LL dm, o; int de; pair<LL, LL> I[N], II[N]; int nn;
LL d1[N], d2[N];

struct edge{
int a, b, w;
void in(int id){
RD(a,b,w); w*=2;
adj[a].PB(id),adj[b].PB(id);
}
} E[N];

void Dijkstra(LL d[], int s){
priority_queue<pair<LL, int>, vector<pair<LL, int> >, greater<pair<LL, int> > > Q;
REP_1(i, n) d[i] = INFF; d[s] = 0, Q.push(MP(0, s));

while (!Q.empty()){
int u = Q.top().se; LL du = Q.top().fi; Q.pop(); if (du != d[u]) continue;

ECH(it, adj[u]){
int v = E[*it].b == u ? E[*it].a : E[*it].b;
int w = E[*it].w;

if (du + w < d[v]){
d[v] = d[u] + w;
Q.push(MP(d[v], v));
}
}
}
}
