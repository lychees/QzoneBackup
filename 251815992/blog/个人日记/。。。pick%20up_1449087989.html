const int N = 50 + 9 ;

const int M = 100 * N* 2 ;

const int LV = 31 ;


LL f[M]; int n, m;

vector< int > pack[LV];


class ClassicProblem {

public :

long long maximalValue(vector < int > c, vector < int > w, vector < int > v, int m) {

n = c.size();

REP(lv, LV) pack[lv].clear();

REP(i, n){

REP(lv, LV) if ((c[i]>>lv)& 1 ){

c[i] -= 1 <<lv;

pack[lv].push_back(i);

}

else {

if (!c[i]) break ;

c[i] -= 1 <<lv;

pack[lv].push_back(i);

c[i] -= 1 <<lv;

pack[lv].push_back(i);

}

}

RST(f); REP(lv, LV){

int mm = min(M, m+ 1 );

if (lv){

int m0 = m& 1 ;

REP(i, mm){

int ii = 2 *i + m0;

if (ii >= mm) ii = mm- 1 ;

f[i] = f[ii];

}

m /= 2 ; mm = min(M, m+ 1 );

}

ECH(it, pack[lv]){

DWN(i, mm-w[*it], 0 ){

checkMax(f[i+w[*it]], f[i] + ((LL)v[*it]<<lv));

}

}

}

return f[ 0 ];

}

};
