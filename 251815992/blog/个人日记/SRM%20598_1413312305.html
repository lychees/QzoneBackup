Summary of the rules

All the vertices in a tree rooted in a vertex that contains a beacon can be uniquely identified if and only if: For each vertex x with c children, at least c − 1 of its sub-trees contain a beacon.

转移的时候用替换法。。。s 表示所有转移必填。。s-t 表示有一个转移不填。。

当且仅当 c == 1 且一定要保证子树必填的时候。。为 s。
其他都是 s-t。。


template<class T> inline bool checkMin(T &a,const T b){return b < a ? a = b, 1 : 0;}
template<class T> inline bool checkMax(T &a,const T b){return a < b ? a = b, 1 : 0;}
const int N = int(51);
vector<string> linked;
int f[N][2], n;

class TPS {
public:

void dfs(int u, int p = -1){
int s = 0, t = 0, c = 0; REP(v, n) if (linked[u][v] == 'Y' && v != p){
dfs(v, u); ++c; s += f[v][1];
checkMax(t, f[v][1]-f[v][0]);
}

if (!c){
f[u][0] = 0;
f[u][1] = 1;
}
else{
f[u][0] = s-t;
f[u][1] = c == 1 ? s : s-t;
}
}

int minimalBeacons(vector <string> linked) {
::linked = linked; n = linked.size(); if (n == 1) return 0;
int z = n; REP(r, n){
dfs(r); checkMin(z, f[r][0]+1);
}
return z;
}
};
