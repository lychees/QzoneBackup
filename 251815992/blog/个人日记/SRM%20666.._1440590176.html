const int N = int(4000) + 9;
Int C[N][N], f[N][4];
int n;

class SumOverPermutations {
public:
int findSum(int n) {

REP(i, n){
C[i][0] = 1;
REP_1(j, i) C[i][j] = C[i-1][j-1] + C[i-1][j];
}

REP(s, 4) f[1][s] = n - count_bits(s);

FOR_1(i, 2, n){
REP(s, 4){
f[i][s] = 0;
f[i][s] += f[i-1][s|_1(0)] * (n-_1(s, 0));
f[i][s] += f[i-1][s|_1(1)] * (n-_1(s, 1));
FOR_1(j, 1, i-2) f[i][s] += f[j][(s&_1(0))|_1(1)] * f[i-1-j][(s&_1(1))|_1(0)] * n * C[i-1][j];
}
}

return f[n][0].val;
}
};
