#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int N = 8;
const int MOD = 1000000007;
int n , m;
LL dp[17][1 << N][1 << N];
vector <pair <int , int> > can[N + 1][1 << N];
// can[i][j]  表示对于每行可以放i列的情况，当前行已经放的状态是j的情况
// first 表示这一行可以放的序列，1表示放个竖的，0的位置，要么被上一行占了，要么可以放横的
// second 表示这样放的话，哪些列是被split的
inline void init () {
for (int i = 1 ; i <= 8 ; i ++) {
for (int j = 0 ; j < 1 << i ; j ++) {
for (int k = 0 ; k < 1 << i ; k ++) {
if (j & k) continue;

int split = 0 , ok = 1;
for (int r = 0 ; r < i ; r ++) {
int now = (1 << r) & (j | k);
int next = (1 << (r + 1)) & (j | k);
if (now) {
split |= 1 << r;
continue;
}
if (r == i - 1 || next) {
ok = 0;
break;
}
r ++;
split |= 1 << r;
}
if (ok) {
if (split & (1 << (i - 1))) split -= 1 << (i - 1);
can[i][j].push_back (make_pair (k , split));
}
}
}
}
}
LL dfs (int row , int state , int split) {
if (row == n) {
return (state == 0 && split == 0) ? 1 : 0;
}
if (dp[row][state][split] != -1) return dp[row][state][split];
LL ret = 0;
for (int i = 0 ; i < (int)can[m][state].size () ; i ++) {
if (row == 0) {
// 第一行的话，必须要有往下竖的
if (can[m][state][i].first)
ret += dfs (row + 1 , can[m][state][i].first , split & can[m][state][i].second);
}
else if (row < n - 1) {
// 中间的，上下都得有
if (state && can[m][state][i].first)
ret += dfs (row + 1 , can[m][state][i].first , split & can[m][state][i].second);
}
else {
// 最后一行，上面必须有
if (state)
ret += dfs (row + 1 , can[m][state][i].first , split & can[m][state][i].second);
}
}
return dp[row][state][split] = ret;
}
int main () {

freopen ("solid.in" , "r" , stdin);
freopen ("solid.out" , "w" , stdout);

init ();
while (cin >> n >> m) {
if (n < m) swap (n , m);
if (n == 1 || n * m % 2 == 1) {
puts ("0");
continue;
}
memset (dp , -1 , sizeof (dp));
printf ("%lld\n" , dfs (0 , 0 , (1 << (m - 1)) - 1));
}
return 0;
}
