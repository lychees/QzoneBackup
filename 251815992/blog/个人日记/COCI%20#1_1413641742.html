#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
using namespace std ;

typedef long long LL ;

const int MAXN = 1000009 ;
const int MAXN2 = 100009 ;

LL N , L ;
int A[MAXN2][26] ;
int K ;

char suf[MAXN] ;

void Init() {
cin >> N ;
gets(suf) ;
gets(suf) ;
L = (LL)strlen(suf) ;
for ( int i = 0 ; i < 26 ; i ++ ) A[0][i] = 0 ;
for ( int i = 1 ; i <= L ; i ++ ) {
A[0][suf[i-1]-'A'] ++ ;
if ( i % 10 == 0 ) {
for ( int j = 0 ; j < 26 ; j ++ ) A[i/10][j] = A[0][j] ;
}
}
for ( int i = 0 ; i < 26 ; i ++ ) A[0][i] = 0 ;
cin >> K ;
}

LL SUM( LL x ) {
return (1LL + x) * x / 2LL ;
}

LL Find( int u , int v ) {
LL ret = 0LL ;
for ( int i = u ; ; i -- ) {
if ( i % 10 == 0 ) {
ret += A[i/10][v] ;
break ;
} else {
if ( suf[i-1]-'A' == v ) ret ++ ;
}
}
return ret ;
}

void Solve() {
for ( int i = 1 ; i <= K ; i ++ ) {
LL a0 ; char c0 ;
cin >> a0 >> c0 ;

LL P1 = SUM(a0-1) % L + 1 ;
LL P2 = SUM(a0) % L + 1 ;

LL ans = (LL)(a0 / L) * Find(L , c0-'A') ;
if ( a0 % L != 0 ) {
if ( P2 > P1 ) ans += Find(P2-1 , c0-'A') - Find(P1-1 , c0-'A') ;
else {
ans += Find(L , c0-'A') - Find(P1-1 , c0-'A') + Find(P2-1 , c0-'A') ;
}
}
cout << ans << "\n" ;
}
}

int main() {
Init() ;
Solve() ;
}
