题意：
给定一个 m 行 n 列的水域。。每个位置的深度为 d[i][j]。。。。
问放置一个 长边不大于 A，短边不大于 B 的箱子，且要求顶端不露出水面（贴着也不行）。。。
最大体积是多少。。。


Analysis：

做法神似 Spoj 912
http://www.shuizilong.com/house/archives/spoj-912-submatrix-of-submatrix/
只不过最里面的函数换成了 POJ 2559
https://gist.github.com/lychees/f4f56d68b7752382df0c


。。。首先考虑水面高度问题。

设池塘的底面积 S = mn。。。箱子的面积 s = ab。。箱子底面距水面的初始深度wei d。
。。箱子的高度为 h。。满足：

Sd + sh (总的水量)
------------ > h
S

整理得：
h < Sd/(S-s) ..
h = [Sd/(S-s] - 1 (上取整）
。。。h 随底面积的增大而增大。。
。。因此我们枚举（箱子的长）a。。。用单调队列。。压缩成一维的问题。。。
。。再用 POJ 2559 （单调栈）求出最大的宽度 b 和 h 即可。。。。

总的复杂度为 O(n3)。。。



。。注意常数。。。
。下面的代码 7s +。。。（如果旋转 90 度的话就会 TLE。。。
。。（避免旋转 90 度的方法是。。每次 checkMin(b, 。。。 的时候判断当前的 a 是否大于 B。。。）




const int N = 509;
int d[N][N]; deque<int> Q[N];
int l[N], r[N]; stack<int> s;
LL z; int m, n, A, B, a, b; LL h;

#define ab (LL)(a*b)
#define mn (LL)(m*n)

void Push(int i, int ii){
if (!Q[i].empty() && Q[i].front() <= ii-a) Q[i].pop_front();
while (!Q[i].empty() && d[ii][i] <= d[Q[i].back()][i]) Q[i].pop_back();
Q[i].PB(ii);
}

void gao(){
REP_1_N(a, min(A, m)){
REP(i, n) CLR(Q[i]); REP(ii, m){
REP(i, n) Push(i, ii); if (ii < a-1) continue;
#define d(i) d[Q[i].front()][i]
//REP(i, n) cout << d(i) << " "; cout << endl;

CLR(s); REP(i, n){
while (!s.empty() && d(i) <= d(s.top())) s.pop();
l[i] = s.empty() ? 0 : s.top() + 1;
s.push(i);
}

CLR(s); DWN(i, n, 0){
while (!s.empty() && d(i) <= d(s.top())) s.pop();
r[i] = s.empty() ? n : s.top();
s.push(i);
}

REP(i, n){
b = min(a <= B ? A : B, r[i] - l[i]), h = ceil(mn*d(i), mn - ab) - 1;
checkMax(z, h*ab);
}
}
}
}

int main(){

#ifndef ONLINE_JUDGE
//freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

RD(A, B, m, n); if (A < B) swap(A, B);
REP_2(i, j, m, n) RD(d[i][j]); gao(), OT(z);
}
