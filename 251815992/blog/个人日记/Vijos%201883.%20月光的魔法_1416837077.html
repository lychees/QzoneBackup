#include <tr1/unordered_map>
const int N = int(3e5) + 9;

PII C[N];
tr1::unordered_map<int, int> P;
int n;


inline int Find(int x){
if (!CTN(P, x)) P[x] = x;
return P[x] == x ? x : P[x] = Find(P[x]);
}

inline void Union(int x, int y){
x = Find(x), y = Find(y); if (x == y) return;
if (rand()&1) P[x] = y; else P[y] = x;
}


int main(){

#ifndef ONLINE_JUDGE
//freopen("kruznice.in.8", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif


RD(n); REP(i, n){
RDD(C[i].se), RD(C[i].fi);
}

sort(C, C+n);

int z = n + 1; REP(i, n){
int l = C[i].se - C[i].fi, r = C[i].se + C[i].fi;
if (Find(l) == Find(r)) ++z;
else Union(l, r);
}

OT(z);
}
