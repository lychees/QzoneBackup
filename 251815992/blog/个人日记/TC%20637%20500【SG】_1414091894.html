表示长度为 n，左边状态 sl，右边状态 sr 时的方案数。。。
（虽然这个是对称的。。并且某侧两个格子都不能填时、相当于长度减一。。但是没必要过分关注。）

const int N = int(2500) + 9;
int F[N][4][4];
int n;

int mex(vector<int> &t){
sort(t.begin(), t.end()); t.erase(unique(t.begin(), t.end()), t.end());
t.push_back(INF); REP(i, t.size()) if (t[i] != i) return i;
}

int f(int n, int sl, int sr){

if (n == 1){
sl |= sr, sr |= sl;
//if (sl == 3) --n;
}
else{
//if (sl == 3) --n, sl = 0;
//if (sr == 3) --n, sr = 0;
}

int &sg = F[n][sl][sr];
if (!~sg){
vector<int> t; REP(o, 2){
if (!_1(sl, o)) t.PB(f(n-1, sl|_1(1-o), sr));
if (!_1(sr, o)) t.PB(f(n-1, sl, sr|_1(1-o)));
}
FOR(i, 1, n-1) REP(o, 2) t.PB(f(i,sl,_1(1-o)) ^ f(n-i-1,_1(1-o),sr));
sg = mex(t);
}
return sg;
}


class PathGame {
public:

sr |= board[o][j] == '#' ? _1(1-o) : 0;
}
sg ^= f(j-i, sl, sr);
}

return sg ? "Snuke" : "Sothe";
