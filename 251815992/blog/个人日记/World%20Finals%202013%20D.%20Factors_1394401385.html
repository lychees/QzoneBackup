const LL M = 1ull<<63; LL C[N][N]; vector<pair<LL, LL> > A;

void dfs(int k=0, int up=N-1, LL f=1, int n=0, LL a=1){
if (k == N) return; if (a!=1) A.PB(MP(f,a)); LL t=1; REP_1(i, up){
t*=P[k]; if (f>M/C[n+i][i] || a>M/t) break;
dfs(k+1, min(i, N-1-n-i), f*C[n+i][i], n+i, a*t);
}
}

LL f(LL n){
return lower_bound(ALL(A), MP((LL)n, (LL)0))->se;
}
