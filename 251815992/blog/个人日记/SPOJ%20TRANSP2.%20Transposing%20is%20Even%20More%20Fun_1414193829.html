const int PMAX = 1000001;
VI P; bitset<PMAX> isC; int phi[PMAX], p[PMAX];
#define ii (i*P[j])
void sieve(){
phi[1] = 1; FOR(i, 2, PMAX){
if (!isC[i]) P.PB(i),phi[i]=i-1,p[i]=i;
for (int j=0;j<SZ(P)&&ii<PMAX;++j){
isC[ii]=1; p[ii]=P[j]; if (!(i%P[j])){
phi[ii] = phi[i]*P[j];
break;
}
else{
phi[ii] = phi[i]*P[j]-phi[i];
}
}
}
}
#undef ii


Int m = 2; Int z = 0; int n, g;

Int f(int d){
return pow(m,g*d)*phi[n/d];
}

void dfs(int n, int d){
if (n == 1){
z += f(d);
}
else{
int c = 1, pn = p[n]; n /= pn; while (p[n] == pn) ++c, n /= pn;
REP(i, c+1){
dfs(n, d);
d *= pn;
}
}
}

OT(int(pow(m, a+b) - z));
}
}
