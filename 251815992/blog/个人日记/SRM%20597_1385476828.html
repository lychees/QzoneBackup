sub-problem:
三种颜色的数量分别是 R,G,B 。。。 问排成一排。使得相邻的颜色不相同的方案数。。


Analysis：
如果没有数目限制的话。。可以直接递推。。有多少种颜色都能做。。。
。。 有数目限制的话。。似乎只能枚举一些东西。。。

。。考虑固定其中一种颜色。。。会形成一些空位。。。共有以下四种可能：

R...R...R  （R-1  空
...R...R...R    或者  R...R...R... (R 个空
...R...R...R...  ( R+1 个空

注意产生 R 个空有两种情况。。。。。考虑插空。。

。。。分奇偶讨论。。。偶数空至少为 2.。。奇数空至少为 1.。。
。枚举偶数空的个数。。设为 e。。。（2^e 。。。
。。则奇数空中需要讨论。。。G, B 哪个会多 1.。。。
设这两个值分别为 og, ob。。。。。显然这组值可以解方程得出。。。
最后组合数乘乘就行了。。。。



/* -- Start from here -- */

const int N = int(1e6) + 9;

Int Fact[N], Factt[N]; Int Binom(int n, int m){
return Fact[n] * Factt[m] * Factt[n-m];
}

class LittleElephantAndBoard {
public:

Int f(int x, int y, int z){
Int res = 0; FOR_1(g, x-1, x+1){
Int cur = 0, ee = 1; for (int e=0;e<=g;++e,ee*=2){
int o = g-e; if ((o+y-z)&1) continue;
int oy = (o+y-z)/2, oz = o-oy; if (oy<0 || oz<0) continue;
int r = y-oy-e; if (r<0) continue;
cur += Binom(g,e)*Binom(r+g-1,r)*Binom(o,oy)*ee;
}
res += cur; if (g == x) res += cur;
}
return res;
}

int getNumber(int M, int R, int G, int B) {
Fact[0] = 1; REP_1(i, N-1) Fact[i] = Fact[i-1] * i;
Factt[N-1] = _I(Fact[N-1]); DWN(i, N, 1) Factt[i-1] = Factt[i] * i;
return f(M-R, M-G, M-B)*2;
}
};
