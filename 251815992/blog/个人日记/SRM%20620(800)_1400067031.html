namespace EQU{
const int N = 20+20+409*40, M = 409;
int A[N][M], AA[N][M];
int n, m;

int Gauss(){
int c,r=0,i; for (c=0;c<m;++c){
FOR_N(i,r,n) if (A[i][c]) break; if (i==n) continue;
if (i != r) FOR_1(j,c,m) swap(A[r][j], A[i][j]);
FOR(i,r+1,n) if (A[i][c]) FOR_1(j,c,m) A[i][j] ^= A[r][j]; ++r;
}
FOR(i, r, n) if (A[i][m]) return 0;
return pow(2, m-r);
}

void init(int nn, map<int, VI>& H){
RST(A), n = 0, m = nn*nn;
REP(i, nn){
REP(j, nn) A[n][i*nn+j] = 1; A[n][m] = 1;
++n;
}
REP(j, nn){
REP(i, nn) A[n][i*nn+j] = 1; A[n][m] = 1;
++n;
}
ECH(it, H){
ECH(jt, it->se) A[n][*jt] = 1;
++n;
}
}
} //using namespace EQU;

map<int, VI> H;

class PerfectSquare {
public:
int ways(vector <int> x) {
sieve(); int n = sqrt(SZ(x)); H.clear();
#define id (i*n+j)
#define d (*it)
REP_2(i, j, n, n) if (x[id] != 1){
ECH(it, P) if (x[id]%d == 0){
int c = 1; x[id] /= d; while (x[id]%d == 0) x[id]/=d,c^=1;
if (c) H[d].PB(id); if (x[id] == 1) break;
}
if (x[id] != 1) H[x[id]].PB(id);
}
EQU::init(n, H);
return EQU::Gauss();
}
};
