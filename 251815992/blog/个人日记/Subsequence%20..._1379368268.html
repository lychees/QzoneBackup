果然还是顺推比较好想好写ww。


const int N = 51;
LL dp[N][N][N]; int p1[26][N], p2[26][N], p3[26][N];

class CountingCommonSubsequences {
public:
long long countCommonSubsequences(string s1, string s2, string s3) {

int n1 = SZ(s1), n2 = SZ(s2), n3 = SZ(s3);
#define p1 p1[c]
#define p2 p2[c]
#define p3 p3[c]
RST(dp); REP(c, 26){
p1[n1] = n1, p2[n2] = n2, p3[n3] = n3;
DWN(i1, n1, 0) p1[i1] = s1[i1] == 'a' + c ? i1 : p1[i1+1];
DWN(i2, n2, 0) p2[i2] = s2[i2] == 'a' + c ? i2 : p2[i2+1];
DWN(i3, n3, 0) p3[i3] = s3[i3] == 'a' + c ? i3 : p3[i3+1];
dp[p1[0]][p2[0]][p3[0]] = 1;
}
#define u dp[i1][i2][i3]
LL res = 0; REP_3(i1, i2, i3, n1, n2, n3) if (u){
res += u; REP(c, 26) dp[p1[i1+1]][p2[i2+1]][p3[i3+1]] += u;
}

return res;
}
};

int n, m;

int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

while (RD(n, m)){
REP_1(i, n) RD(A[i]); REP_1(i, m) RD(B[i]);
REP_2_1(i, j, n, m){
if (A[i] == B[j]) dp[i][j] = dp[i-1][j-1] + 1;
else dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
}
OT(dp[n][m]);
}
}

SCS (LCS 的对偶问题

const int N = 109;
char A[N], B[N]; int dp[N][N]; LL ss[N][N];
int n, m;

int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

REP(i, N) ss[0][i] = ss[i][0] = 1;

Rush{
n = strlen(RS(A+1)), m = strlen(RS(B+1));

REP_2_1(i, j, n, m){
if (A[i] == B[j]){
dp[i][j] = dp[i-1][j-1] + 1;
ss[i][j] = ss[i-1][j-1];
}
else{
dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
ss[i][j] = (dp[i][j] == dp[i-1][j] ? ss[i-1][j] : 0) + (dp[i][j] == dp[i][j-1] ? ss[i][j-1] : 0);
}
}

printf("Case #%d: %d %lld\n", ++Case, n+m-dp[n][m], ss[n][m]);
}
}

证明：
求最长公共子序列时如果 A[i] == B[j]，
。。。取 dp[i-1][j-1]+1 即可。。且此时 ss[i-1][j] 和 ss[i][j-1] 的方案数 。。包括在 ss[i-1][j-1] 之中。
http://www.cnblogs.com/staginner/archive/2011/12/06/2278307.html

这个性质在 DP 在求 LCS 长度的时候作用不是很明显。。。在求方案数的时候就至关重要了。。。。
