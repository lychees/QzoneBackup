const int N = int(2e6) + 9, Z = 26;
char buf[N];

namespace BIT{
int C[N], n;
void Add(int x, int d){
for (;x<=n;x+=low_bit(x)) C[x] += d;
}
int Sum(int x){
int res = 0; for (;x;x^=low_bit(x)) res += C[x];
return res;
}
int Sum(int l, int r){
return Sum(r)-Sum(l-1);
}
} using namespace BIT;

namespace ACM{
int trans[N][Z], fail[N], faill[N], cnt[N]; int id[N]; int pos[N];
int Q[N], u, cz, op, tot; int vis[N], vis_tag;
VI adj[N]; int L[N], R[N];

inline int new_node(){
//RST(trans[tot]), fail[tot] = 0;
return tot++;
}

#define v trans[u][c]
#define f trans[fail[u]][c]

void dfs(int u = 0){
L[u] = ++n; ECH(it, adj[u]) dfs(*it); R[u] = n;
}


void Build(){
cz = op = u = 0; REP(c, Z) if (v) Q[op++] = v;
while (cz < op){
u = Q[cz++]; REP(c, Z) if (v) fail[Q[op++] = v] = f, faill[v] = id[f] ? f : faill[f];
else v = f;
}
}

int u = 0; RS(buf); REP_S(cur, buf){

}

return u;
}

REP_1_C(i, RD()){

}
