250：
给定一棵边权树，求最多可以选多少个点。。使得这些点的距离两两相等。


。。。首先特判 n<=2 的情况。。

对于 n > 2。。。

。答案至少是 2.。。对于更高的答案。。一定是以某个点为中心。。然后这样。。。
1
2   3   4   5
。。但是注意这些点到 1 的路径不能重合。。（否则有 lca 。。。不能保证两两 xxxxx）

我的做法是枚举中心。。。然后对每个子树树 DP 出一个 set<int>。。。再用这个 set<int> 维护一个 map<int, int>。。。

const int N = 109;
int d[N]; bool vis[N];

vector<pair<int, int > > adj[N];
void dfs(int u = 1){
vis[u] = 1;
ECH(it, adj[u]) if (!vis[v]){
d[v] = d[u] + it->se;
dfs(v);
}
}

set<int> dfs2(int u){
set<int> A; A.insert(0); vis[u] = 1; ECH(it, adj[u]) if (!vis[v]){
set<int> B = dfs2(v);
ECH(jt, B) A.insert(*jt + it->se);
}
return A;
}


template<class T> inline T& checkMax(T &a,const T b){if (a<b) a=b;return a;}
template<class T> inline void RST(T &A){memset(A, 0, sizeof(A));}
#define REP_1(i, n) for (int i=1;i<=n;++i)
class Egalitarianism3 {
public:
int maxCities(int n, vector <int> a, vector <int> b, vector <int> len) {

if (n == 1) return 1;

//if (n == 2) return 2;
RST(d);

int ans = 2;

REP_1(i, n) adj[i].clear();

REP(i, n-1){
adj[a[i]].push_back(make_pair(b[i], len[i]));
adj[b[i]].push_back(make_pair(a[i], len[i]));
}

dfs();

REP_1(i, n){
RST(vis); vis[i] = 1;
map<int, int> H; ECH(it, adj[i]){
set<int> t = dfs2(v);
ECH(jt, t) ++H[*jt + it->se];
}

}
