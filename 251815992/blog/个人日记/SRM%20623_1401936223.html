template<class T, class C> int LIS(int n, const vector<T>& a, C cmp){
if (!n) return 0;
vector<T> b; b.PB(a[0]); FOR(i, 1, n){
if (cmp(b.back(), a[i])) b.PB(a[i]);
else {
b[lower_bound(ALL(b), a[i], cmp) - b.begin()] = a[i];
}
}
return b.size();
}


class CatchTheBeat {
public:
int maxCatched(int n, int x0, int y0, int a, int b, int c, int d, int mod1, int mod2, int offset) {

VII I; I.resize(n); I[0] = MP(x0, y0); FOR(i, 1, n){
I[i] = MP(((LL)I[i-1].fi*a+b)%mod1,((LL)I[i-1].se*c+d)%mod2);
}

int nn = 0; REP(i, n){
I[i].fi -= offset; if (I[i].se < abs(I[i].fi)) continue;
int x = I[i].fi, y = I[i].se;
I[nn++] = MP(y-x, x+y);
}

I.resize(n = nn); SRT(I); VI A; REP(i, n) A.PB(I[i].se);
return LIS(n, A, less_equal<int>());
return 0;
}
};
