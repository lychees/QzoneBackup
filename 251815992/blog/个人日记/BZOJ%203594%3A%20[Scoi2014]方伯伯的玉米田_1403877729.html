1. 朴素

修改操作一定是一些后缀。。。否则结果不会更优...

f[i][j] 。。
表示：前 i  个数，第 i 个数不删，使用了【至多】 j 次+操作，至多可以保留多少株。
转移。。枚举 i' 。。表示 (i', i) 这段全删。。。

若 h[i'] <= h[i] 则可以直接从 f[i'][j] 转移。
否则，则要求 h[i'] + j' <= h[i] + j。。。
。。取最大的 j' = h[i] + j - h[i']
。。于是可以暴力 O(n2m)。。


//}/* .................................................................................................................................. */

const int N = int(1e4) + 9, M = 509;
int f[N][M], h[N];
int n, m;

int main(){

#ifndef ONLINE_JUDGE
freopen("in.txt", "r", stdin);
//freopen("out.txt", "w", stdout);
#endif

RD(n, m);

#define u f[i][j]
#define v f[ii]

int res = 0; REP_1(i, n){
RD(h[i]); FOR_1(j, 0, m){

FOR(ii, 1, i){
if (h[i] >= h[ii]) checkMax(u, v[j]);
else if (h[i]+j >= h[ii]) checkMax(u, v[h[i]+j-h[ii]]);
}

checkMax(res, u += 1);
}
}
OT(res);
}
