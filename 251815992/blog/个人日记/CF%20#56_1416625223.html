const int M = int(1e7) + 9;
const int N = int(1e6) + 9;

namespace DSU{ // Disjoint Set Union
int P[M], R[M];
inline void Make(int x){
P[x] = x, R[x] = 0;
}
inline int Find(int x){
return P[x] == x ? x : P[x] = Find(P[x]);
}
inline void Unionn(int x, int y){
if (R[x] == R[y]) ++R[x];
else if (R[x] < R[y]) swap(x, y);
P[y] = x;
}
inline void Union(int x, int y){
x = Find(x), y = Find(y), Unionn(x, y);
}
inline void Init(){
REP(i, M) Make(i);
}
} using namespace DSU;


bool X[2*M]; int A[N];
int n;

const int PMAX = 1000;
VI P; bitset<PMAX> isC;
#define ii (i*P[j])
void sieve(){
FOR(i, 2, PMAX){
if (!isC[i]) P.PB(i);
for (int j=0;j<SZ(P)&&ii<PMAX;++j){
isC[ii]=1; if (!(i%P[j])) break;
}
}
}
#undef ii

const int N = int(1e2) + 9;

int f(int n, int p){
int z = 0; while (n % p == 0) ++z, n /= p;
return z;
}

SI Fact; MII fact(int n){
MII z; ECH(p, P){
if (*p > n) break;
if (n % *p == 0){
int t = 1; n /= *p;
while (n % *p == 0) ++t, n /= *p;
z[*p] = t;
}
}
if (n != 1) z[n] = 1;
return z;
}

struct edge{
int to, gcd, lcm;
void in(){
RD(gcd, lcm); MII t = fact(gcd); ECH(it, t) Fact.insert(it->fi);
t = fact(lcm); ECH(it, t) Fact.insert(it->fi);
}
};

vector<edge> adj[N];
int ans[N], cur[N]; bool vis[N], viss[N], visss[N]; int Time; int Gcd[N][N], Lcm[N][N];
int n, m;

#define v it->to

bool dfs(int u){
vis[u] = 1; ECH(it, adj[u]) if (!vis[v]){
int a = Gcd[u][v], b = Lcm[u][v]; cur[v] = a+b-cur[u];
if ((a != cur[u] && b != cur[u]) || !dfs(v)) return 0;
}
return 1;
}

bool ck(int p){

RST(Gcd, Lcm); REP_1(u, n){
cur[u] = 0; ECH(it, adj[u]){
Gcd[u][v] = f(it->gcd, p);
Lcm[u][v] = f(it->lcm, p);
if (Lcm[u][v] < Gcd[u][v]) return 0;
}
}

RST(vis); REP_1(u, n) if (!vis[u]){
ECH(it, adj[u]){
int a = Gcd[u][v], b = Lcm[u][v];
cur[u] = a; CPY(viss, vis); if (!dfs(u)){
cur[u] = b; CPY(vis, viss); if (!dfs(u)) return 0;
}
break;
}
}

return 1;
}
