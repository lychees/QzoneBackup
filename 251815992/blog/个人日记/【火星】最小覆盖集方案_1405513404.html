König's theorem 说的是。。最小覆盖集 = 最大匹配。。。

。。。首先必要性显然。。（覆盖  >= 匹配  ）
。。至于取等号的时候 wiki 上描述了一种构造性证明。

。。。这段证明不仅说明了这个定理的正确性。。
。。。也提供了一种求最大匹配的方法。（反复找增广路）。
。。和构造最小覆盖集的方法。。（找交替路。。取所有夹在中间的点）


。。简而言之。就是下面这幅图。。。。




。。。所有奇数层的点是最小覆盖集（对应的偶数层就是独立集。。）
。。这样就可以 AC  sensor 了。。。
https://gist.github.com/lychees/7cf3848c3845f9d6bc53


。。 wiki 原始版本的构造是逐层 bfs() 的。

“If there is no such vertex, but there remain vertices not included in any previously-defined set S k , arbitrarily choose one of these and let S 2 j +1 consist of that single vertex.”。。

。。注意这种【 孤立匹配】的情况。。。随便取一个点即可。。不需要再扔到队列中去。。。
。尽管如此 。。。bfs() 写起来还是十分别扭。。还是 dfs() 自然许多。。。

const int N = 109;

VI adj[N], adjy[N]; int vx[N], vy[N]; int py[N], px[N];
int n; VI res;

#define y (*it)
#define vis vy[y]
#define p py[y]
bool dfs(int x){
ECH(it, adj[x]) if (!vy[y]){
vis = 1;
if (!~p || dfs(p)){
p = x;
return 1;
}
}
return 0;
}
#undef p

void dfs_y(int, int);

void dfs_x(int x, int bj){
vx[x] = bj, bj ^= 1; if (bj){
ECH(it, adj[x]) if (!~vy[y]) dfs_y(y, bj);
}
else{
#undef y
int y = px[x]; dfs_y(y, bj);
}
}

void dfs_y(int y, int bj){
vy[y] = bj, bj ^= 1; if (bj){
#define x (*it)
ECH(it, adjy[y]) if (!~vx[x]) dfs_x(x, bj);
}
else{
#undef x
int x = py[y]; dfs_x(x, bj);
}
}




VI X, Y; Po P[N]; int d;
int nn;

int main(){

#endif

RD(nn, d); d*=d; REP(i, nn) P[i].in(); res.PB(0);

REP_2(jj, ii, nn, jj) if (dist2(P[ii], P[jj]) <= d){

int dd = dist2(P[ii], P[jj]);

CLR(X, Y); REP(i, nn) if (i != ii && i != jj && dist2(P[ii], P[i]) <= dd && dist2(P[jj], P[i]) <= dd){
if (dett(P[ii], P[jj], P[i]) == 1) X.PB(i);
else Y.PB(i);
}

fill(py, py+SZ(Y), -1);

REP(i, SZ(X)) adj[i].clear(); REP(i, SZ(Y)) adjy[i].clear();
REP(i, SZ(X)) REP(j, SZ(Y)) if (dist2(P[X[i]], P[Y[j]]) > dd) adj[i].PB(j), adjy[j].PB(i);

int cur = SZ(X)+SZ(Y)+2; FLC(py, -1); REP(i, SZ(X)){
fill(vy, vy+SZ(Y), 0); if (dfs(i)) cur--;
}

if (cur > SZ(res)){

fill(px, px+SZ(X), -1); REP(i, SZ(Y)) if (~py[i]) px[py[i]] = i;
fill(vx, vx+SZ(X), -1); fill(vy, vy+SZ(Y), -1);

REP(x, SZ(X)) if (!~px[x]) dfs_x(x, 0);
REP(y, SZ(Y)) if (!~py[y] && !~vy[y]) dfs_y(y, 0);
res.clear(); res.PB(ii), res.PB(jj);
REP(i, SZ(X)) if (vx[i] == 0) res.PB(X[i]);
REP(i, SZ(Y)) if (vy[i] <= 0) res.PB(Y[i]);
assert(cur == SZ(res));
}
}

OT(SZ(res));
ECH(it, res) printf("%d ", *it+1);
}
