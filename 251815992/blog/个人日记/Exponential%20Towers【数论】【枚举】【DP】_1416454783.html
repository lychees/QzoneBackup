给定一个可以表达成 height = 3 的 Exponential Towers 的数 n = a^{b^c}。。。
问其可以表达成多少 height >= 3 的形式。。。

————————————————


感谢数一。。。

首先观察 height = 2 的情况。。。 设 n = a1^b1 = a2^b2 = ... a3^b3。。。
容易看出。。最大的 b = 素数分解后所有指数的最大公约数 。。。
设之为 g。。那么有标准形式 n = x^g。。设 g= uv。。那么有 n = (x^u)^v 。。。因此方案数就是 g 的约数个数。。。

可见我们需要关心的是 g 的值。。。x 的值是不重要的。。

考虑 height = 3 的情况。。n = a^{b^c} 。。。
先将 a 分解成标准形式 a = x^g。。。
有 n = (x^g)^{b^c} = x^{gb^c}

同样 x 是不需要关心的。。。
只要考察 gb^c

。设 G = gb^c = uv^w。。。
这里我们只需要枚举 w。。。计算出 v 的方案数即可。。。
虽然 G 本身可能特别大。。。好在其对应的素数分解是容易得到的。。。

假设 G = \Pi p^e 。。。。 容易得出这个方案数为
\Pi {e/w + 1} - 1 // 定义中要求 v 不能等于 1.。。

对于 height > 3 的情况。。只需要将 w 进一步转写即可。。。由于这个下降的很快。。可以记忆化搜索。

const int N = int(1e3) + 9;

//线性素筛。。
const int PMAX = 10000;
VI P; bitset<PMAX> isC;
#define ii (i*P[j])
void sieve(){
FOR(i, 2, PMAX){
if (!isC[i]) P.PB(i);
for (int j=0;j<SZ(P)&&ii<PMAX;++j){
isC[ii]=1; if (!(i%P[j])) break;
}
}
}
#undef ii


MII fact(int n){
MII z; ECH(p, P) if (n % *p == 0){
int t = 1; n /= *p; while (n%*p == 0) n /= *p, ++t;
z[*p] = t;
}
if (n>1) z[n] = 1;
return z;
}

const int WMAX = 32 * 8192;

int F[WMAX];
int f(int n){
int &res = F[n];
if (res == 0){
res = 1; MII t = fact(n);
int g = 0; ECH(it, t) g = gcd(g, it->se);
//cout << n << " " << g << endl;
for (int i=1;i<g;++i) if (g % i == 0){
res += f(g/i);
}
}
return res;
}



int main(){

#ifndef ONLINE_JUDGE

//freopen("out.txt", "w", stdout);
#endif

sieve();

int a, b, c; while (~scanf("%d^%d^%d", &a, &b, &c)){
MII fa = fact(a); int g = 0; ECH(it, fa) g = gcd(g, it->se);
MII fb = fact(b); ECH(it, fb) it->se *= c;
MII fg = fact(g); ECH(it, fg) fb[it->fi] += it->se;

int wmax = 0; ECH(it, fb) checkMax(wmax, it->se);

LL res = 0; FOR_1(w, 2, wmax){
LL z = 1; ECH(it, fb) z *= it->se/w+1; --z;
res += z * f(w);
}
printf("%lld\n", res);
}
}
