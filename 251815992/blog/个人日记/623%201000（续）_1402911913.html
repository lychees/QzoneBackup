int ff(int a){

if (A-a > BB) return 4; if (A+a < B) return -4;
（四分全取的情况。。。
if (!a) return A == BB ? 2 : 0;  （A 不存在讨论的情况。。
if (A-a < B) return A<B ? -2 : 0; （B 可以取决策 0 的情况。。。

if (A-a == BB) return 3; if (A+a == BB) return 1;  （3 和 1 的唯一情况。。。
if (A+a == B) return -2; if (A+a < BB) return 0; （剩下的情况。。。
assert(A+a > BB);
return 2;
}

。。。该函数就是这个题的全部。。

abs(A-B) 和 abs(A-BB)。。。（以及其左右）

const int N = 2009;
int cards[N], bonus[N], n;
int A, B, BB;

int ff(int a){

if (A-a > BB) return 4; if (A+a < B) return -4;

if (!a) return A == BB ? 2 : 0;
if (A-a < B) return A<B ? -2 : 0;

if (A-a == BB) return 3; if (A+a == BB) return 1;
if (A+a == B) return -2; if (A+a < BB) return 0;
assert(A+a > BB);
return 2;
}

vector<pair<int, PII> > I;

void _Add(int a, int j){
if (a < 0 || a > A) return;
I.PB(MP(a, MP(j, ff(a))));
}

void Add(int a, int j){
_Add(a-1, j),_Add(a, j),_Add(a+1, j);
}


int f(int i){
I.clear(), A = cards[i];

int z = 0; REP(j, n) if (i != j){
B = cards[j], BB = B*2, z += ff(0), bonus[j] = ff(0);
Add(A-B, j), Add(B-A, j), Add(A-BB, j), Add(BB-A, j);
}


SRT(I); int zz = z; for (int i=0;i<SZ(I);){
do{
int id = I[i].se.fi, bo = I[i].se.se, d = bo-bonus[id];
zz += d, bonus[id] = bo, ++i;
}while (i<SZ(I) && I[i-1].fi == I[i].fi);

checkMax(z, zz);
}

return z;
}
