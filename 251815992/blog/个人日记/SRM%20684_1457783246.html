考虑 ??? 里面的内容。。。
有趣的是。。这里是一串约数序列。。。
由于每次至少 *2。。因此这个序列不会太长。。

这样 dp 复杂度只有 O(nlogn)。。。
预处理复杂度，O(nlogn sqrt(n))



//}/* .................................................................................................................................. */

const int N = int(5e4) + 9, LV = 20;

// bad[n] = k bad[n-1] + good[n-2]*? - good[n-3]*?? ...

Int good[N], bad[N], divseq[N], dp[LV][N];

class DivFree {
public:
int dfcount(int n, int k) {

RST(good, bad, divseq, dp);

REP_1(d, k) dp[0][d] = 1;

FOR(lv, 1, LV){
REP_1(d, k) for (int j = d+d; j <= k; j += d) dp[lv][j] += dp[lv-1][d];
divseq[lv] = accumulate(dp[lv], dp[lv]+N, Int(0));
}

bad[0] = 0; good[0] = 1;

REP_1(i, n){
bad[i] = bad[i-1] * k;
FOR(lv, 1, LV){
if (i-1-lv < 0) break;
if (lv & 1) bad[i] += good[i-1-lv] * divseq[lv];
else bad[i] -= good[i-1-lv] * divseq[lv];
}
good[i] = pow(k, i) - bad[i];
}

return good[n];
}
};
