决策单调性。。
用 deque + 二分查找维护决策区间，。。复杂度 O(nlogn)。。
复杂度比斜率优化差，不过适用范围更广。

（deque 所维护状态， pair<int, int>：
（first：表示决策点。
（second：该决策点最远可以影响的右端点位置。（左端点位置需要从前一个状态里取。。（所以多保留一个空状态！

const int N = int(5e4) + 9;

LL f[N], s[N]; PII q[N]; int cz, op;
int n, L;

LL calc(int j, int i){
return f[j] + sqr(s[i]-L-s[j]);
}

int find(int l, int r, int o, int n){
while (l < r){
int m = l + r + 1 >> 1;
if (calc(n, m) < calc(o, m)) r = m - 1;
else l = m;
}
return l;
}

int main(){

#ifndef ONLINE_JUDGE

RD(n, L); REP_1(i, n) s[i] = s[i-1] + RD();
REP_1(i, n) s[i] += i; ++L;

q[1] = {0, n}; cz = 1, op = 1; REP_1(i, n){
f[i] = calc(q[cz += (q[cz].se < i)].fi, i);
if (cz > op || calc(i, n) < calc(q[op].fi, n)){
while (cz <= op && calc(i, q[op-1].se+1) < calc(q[op].fi, q[op-1].se+1)) --op;
q[op].se = find(q[op-1].se+1, q[op].se, q[op].fi, i);
q[++op] = {i, n};
}
}

OT(f[n]);
}
