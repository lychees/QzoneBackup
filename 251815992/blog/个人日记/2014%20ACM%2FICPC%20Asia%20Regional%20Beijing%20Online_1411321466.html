const int N = int(3e4) + 9, M = 2 * N;
int hd[N], suc[M], to[M], ww[N]; map<string, int> H;
int ex[N], ey[N], m;
int L[N], R[N], nn, n;

#define aa to[i^1]
#define bb to[i]
#define v bb
#define w ww[i/2]

namespace ST{

int A[N], T[4*N]; bool r0[4*N];
int a, b;

#define lx (x<<1)
#define rx (lx|1)
#define ml (l+r>>1)
#define mr (ml+1)
#define rt 1, 1, n
#define lc lx, l, ml
#define rc rx, mr, r

void upd(int x){
T[x] = T[lx] + T[rx];
}
void rev(int x, int l, int r){
r0[x] ^= 1;
T[x] = (r-l+1) - T[x];
}

void rls(int x, int l, int r){
if (r0[x]){
rev(lx, l, ml), rev(rx, mr, r);
r0[x] = 0;
}
}

void Build(int x, int l, int r){
r0[x] = 0;
if (l == r){
T[x] = A[l];
}
else{
Build(lc), Build(rc);
upd(x);
}
}

void Reverse(int x, int l, int r){
if (b < l || r < a) return;
if (a <= l && r <= b){
rev(x, l, r);
}
else{
rls(x, l, r);
Reverse(lc), Reverse(rc);
upd(x);
}
}

void Reverse(int l, int r){
a = l, b = r;
Reverse(rt);
}

} using namespace ST;


void dfs(int u = 1, int c = 0){
L[u] = ++nn, A[nn] = c; REP_G(i, u) if (!L[v]){
dfs(v, c^w);
}
R[u] = nn;
}
