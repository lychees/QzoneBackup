const int PMAX = 1;
VI P; bitset<PMAX> isP;
void sieve(){
FOR(i, 2, PMAX){
if (!isP[i]) P.PB(i);
for (int j=0;j<SZ(P)&&i*P[j]<PMAX;++j){
isP[i*P[j]]=1; if (!(i%P[j])) break;
}
}

莫比乌斯反演
\paragraph{约数形式}
$$g(n)=\sum_{d\,\mid \,n}f(d)\quad\text{for every integer }n\ge 1$$
$$f(n)=\sum_{d\,\mid\, n}\mu(d)g(n/d)\quad\text{for every integer }n\ge 1$$
\paragraph{倍数形式}
$$g(d)=\sum_{d\,\mid \,n}f(n)\quad\text{for every integer }n\ge 1$$
$$f(d)=\sum_{d\,\mid\, n}\mu(n)g(n)\quad\text{for every integer }n\ge 1$$
