500：

。。我们要证明存在一种最佳顺序。。。也就是。如果按照这种顺序执行都不行。。那么一定不存在其他顺序可行。。。

考虑两个物件 (duration, refuel)
1 = (d1, r1),
2 = (d2, r2)。。


考虑两种方案。。
1 -> 2:
A = max(d1, d2+d1-r1)
2 -> 1:
B = max(d2, d1+d2-r2)。。


若 A 方案优于 B 方案。。则 A < B 。。
由于 d-r 严格递增。。。
则必须满足

d2 + d1 - r1 < d1 + d2 - r2。
既 r1 > r2

因此按照 r 排序即可。。。（注意背包的顺序是和这个相反的、、、

VI x, y, xx, yy;

#define k(i) ((y[i]-y[i-1])/(x[i]-x[i-1]))
#define kk(i) ((yy[i]-yy[i-1])/(xx[i]-xx[i-1]))

void add(int n, int t, int d){

CLR(xx, yy); REP(i, SZ(x)){
if (i && x[i-1]<t && x[i]>t) xx.PB(t), yy.PB(y[i-1]+k(i)*(t-xx[i-1]));
xx.PB(x[i]), yy.PB(y[i]+abs(t-x[i]));
}

/*REP(i, SZ(xx)){

int i = 0; if (d){
x = xx, y = yy; CLR(xx, yy);
while (i!=SZ(x)-1 && y[i+1]<=y[i]) xx.PB(x[i]-d), yy.PB(y[i++]);
xx.PB(x[i]-d), yy.PB(y[i]), xx.PB(x[i]+d), yy.PB(y[i++]);
while (i<SZ(x)) xx.PB(x[i]+d), yy.PB(y[i++]);
}

CLR(x, y); REP_N(i, SZ(xx)) if (xx[i] >= 0) break;
if (xx[i] > 0) x.PB(0), y.PB(yy[i-1]+kk(i)*-xx[i-1]); FOR_N(i, i, SZ(xx)){
if (xx[i] > n) break;

}
if (x.back() != n) x.PB(n), y.PB(yy[i-1]+kk(i)*(n-xx[i-1]));
}

int f(int n, VI t, VI d){

return n*SZ(t)-*min_element(ALL(y));
}
