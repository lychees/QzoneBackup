DB r; VP P; VF I; int n;

Line L(DB x){
return Line(Po(0, 0), Po(cos(x), sin(x)));
}

DB S(Po p0, Po p1, Po p2){
DB a = dist(p0, p1), b = dist(p0, p2), c = dist(p1, p2), p = (a+b+c) / 2;
return sqrt(p*(p-a)*(p-b)*(p-c));
}

DB f(Seg s, int l, int r){
DB z = 0; FOR(i, l, r){
DB d = (s*L((I[i]+I[i+1])/2)).len2();
z += sgn(d, ::r) < 0 ? S(Po(), s*L(I[i]), s*L(I[i+1])) : ::r*(I[i+1]-I[i])/2;
}
return z;
}

DB A = (P[i+1]-P[i]).len2(), B = 2*((P[i].x-P[i+1].x)*P[i+1].x+(P[i].y-P[i+1].y)*P[i+1].y), C = P[i+1].len2()-r, D = sqr(B) - 4*A*C;
