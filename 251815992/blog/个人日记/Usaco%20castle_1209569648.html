原题目的意思是给你一张城堡的平面图..~
要求求出面积最大的一个房间，房间的个数
以及打破一扇墙壁所能获得的最大面积数～

{5
9
16
4 1 E} 样例数据的输出～

一个简单的思路是对整副地图作一次floodfill～
得出每块部分的面积～再一次扫描求出打破一扇墙壁所能获得的最大面积～

= -由于考虑到时无向图，求连通子图的方法可以使用并查集～
这道题
ufs=^rec;
rec=record
n:int;
c:int;
p:ufs;
k:int;
end;
s:array[1..50,1..50] of ufs;
数据结构的设计：～并查集的四个域名分别是
n:按秩合并用 c:记录方块的数量，以便求出题目中的信息
p:指针       k:用来存储地图信息

接下来我们的工作就很简单啦～通过一次扫描合并联通的区域我们可以获得第一问和第二问的答案
for a:=1 to m do
for b:=n downto 1 do
if (如果相邻两块区域不联通) and (之间有墙) and (位置合法) then  考虑是否基于推倒墙壁可以获得更优的方案

于是本题可以得到解决～～




附：My program~
{ID: Lychees
PROG: castle
LANG: PASCAL
}
type int=integer;
ufs=^rec;
rec=record
n:int;
c:int;
p:ufs;
k:int;
end;
var input,output:text;
a,b,c:int;
n,m:int;
room,max:int;
s:array[1..50,1..50] of ufs;


procedure make(var x:ufs);
begin
new(x);x^.k:=0;x^.p:=x;x^.n:=0;x^.c:=1;
end;
function find(var x:ufs):ufs;
begin
if x^.p<>x then x^.p:=find(x^.p);
exit(x^.p);
end;
procedure union(x,y:ufs);
begin
x:=find(x);y:=find(y);if x=y then exit;
if x^.n<y^.n then begin x^.p:=y^.p;inc(y^.c,x^.c);if y^.c>max then max:=y^.c; end
else begin y^.p:=x;inc(x^.c,y^.c); if x^.n=y^.n then inc(x^.n);if x^.c>max then max:=x^.c;  end;
dec(room);
end;
function North(x:ufs):boolean;
begin
if x^.k and 2 =0 then exit(true);
exit(false);
end;
function South(x:ufs):boolean;
begin
if x^.k and 8 = 0 then exit(true);
exit(false);
end;
function West(x:ufs):boolean;
begin
if x^.k and 1 = 0 then exit(true);
exit(false);
end;
function East(x:ufs):boolean;
begin
if x^.k and 4 =0  then exit(true);
exit(false);
end;

{-------------------------------------}
procedure load;
begin
assign(input,'castle.in');assign(output,'castle.out');reset(input);rewrite(output);
readln(input,m,n);room:=m*n;max:=1;
for a:=1 to n do
begin
for b:=1 to m do
begin
make(s[a,b]);read(input,s[a,b]^.k);
end;
readln(input)
end;
close(input);
end;
procedure main;
begin
for a:=1 to n do
begin
for b:=1 to m do
begin
if (North(s[a,b])) and (a>1) then union(s[a,b],s[a-1,b]);
if (South(s[a,b])) and (a<n) then union(s[a,b],s[a+1,b]);
if (West(s[a,b])) and (b>1) then union(s[a,b],s[a,b-1]);
if (East(s[a,b])) and (b<n) then union(s[a,b],s[a,b+1]);
end;
end;
end;
procedure print;
var t,d:integer;
x,y:integer;
z:char;
begin
writeln(output,room);
writeln(output,max);d:=0;
for a:=1 to m do
begin
for b:=n downto 1 do
begin
if not (North(s )) and (b>1) and (find(s )^.p<>find(s )^.p) then
begin
t:=find(s )^.c+find(s )^.c;
if t>d then begin d:=t;z:='N';x:=b;y:=a end;
end;
if not (East(s )) and (a<m) and (find(s )^.p<>find(s )^.p) then
begin
t:=find(s )^.c+find(s )^.c;
if t>d then begin d:=t;z:='E';x:=b;y:=a end;
end;
end;
end;
writeln(output,d);
writeln(output,x,' ',y,' ',z);
close(output);
end;
begin
load;
main;
print;
end.
